 13 // Use a native Object.create if found.  Otherwise, simulate one
 14 function ObjectCreate(proto) {
 15 	if (Object.create) return Object.create(proto);
 16 	var ctor;
 17 	if (proto) {
 18 		ctor=createFunction();
 19 		ctor.prototype=proto;
 20 		return new ctor();
 21 	}
 22 	return {};
 23 }

 25 /**
 26  * Takes a constructor written according to the following pattern and
 27  * updates its prototype chain to inherit from the given superCtor's
 28  * prototype.
 29  * <pre>
 30  * function BaseClass() {
 31  * }
 32  * BaseClass.prototype={
 33  *   // base class properties
 34  * };
 35  *
 36  * function DerivedClass() {
 37  *   BaseClass.call(this);
 38  * }
 39  * DerivedClass.prototype={
 40  *   // derived properties
 41  * };
 42  * inherits(DerivedClass, BaseClass)
 43  * </pre>
 44  * 
 45  * Note that the original DerivedClass.prototype is discarded as part of this
 46  * operation, being used solely to initialize the newly created inherited
 47  * prototype.
 48  * @public
 49  */
 50 function inherits(ctor, superCtor) {
 51 	var newProto=ObjectCreate(superCtor.prototype);
 52 	ObjectCopy(ctor.prototype, newProto);
 53 	ctor.prototype=newProto;
 54 }

 56 /**
 57  * Copy all own properties from src to dest (creating if needed)
 58  * @return dest
 59  */
 60 function ObjectCopy(src, dest) {
 61 	dest=dest||{};
 62 	if (src) {
 63 		for (var k in src) {
 64 			if (src.hasOwnProperty(k))
 65 				dest[k]=src[k];
 66 		}
 67 	}
 68 	return dest;
 69 }

 85 /**
 86  * If called without a name, returns the object of event lists.
 87  * If called with a name, returns the event list for the given
 88  * name.  Always allocates objects as necessary.
 89  *
 90  * @private
 91  * @methodOf nanomaps.EventEmitter
 92  * @name _evt
 93  */
 94 EventEmitterMethods._evt=EventEmitterMethods.listeners=function(name) {
 95 	var hash=this.__evt, list;
 96 	if (!hash) {
 97 		this.__evt=hash={};
 98 	}
 99 	list=hash[name];
100 	if (!list) list=hash[name]=[];
101 	return list;
102 };

104 /**
105  * Add an event listener that will be invoked on every following occurence of the
106  * named event.  The listener is invoked with <b>this</b> equal to the instance
107  * that raised the event and arguments specific to the event type.
108  * 
109  * The method "on" is a synonym of addListener.
110  * 
111  * @public
112  * @methodOf nanomaps.EventEmitter.prototype
113  * @name addListener
114  * @param {string} event Event name to add listener to
115  * @param {function} listener Callback to be invoked on event
116  */
117 EventEmitterMethods.addListener=EventEmitterMethods.on=function(event, listener) {
118 	this._evt(event).push(listener);
119 };
120 
121 /**
122  * Add an event listener that will be invoked on the very next occurence of the
123  * named event.  The listener is invoked with <b>this</b> equal to the instance
124  * that raised the event and arguments specific to the event type.
125  *
126  * @public
127  * @methodOf nanomaps.EventEmitter.prototype
128  * @name once
129  * @param {string} event Event name to add listener to
130  * @param {function} listener Callback to be invoked on event
131  */
132 EventEmitterMethods.once=function(event, listener) {
133 	this._evt(event+'$once').push(listener);
134 };
135 
136 /**
137  * Remove a previously added listener.  Does nothing if the given listener
138  * is not found.
139  
140  * @public
141  * @methodOf nanomaps.EventEmitter.prototype
142  * @name removeListener
143  * @param {string} event Event name to add listener to
144  * @param {function} listener Previously added callback to remove
145  */
146 EventEmitterMethods.removeListener=function(event, listener) {
147 	removeFromList(this._evt(event));
148 	removeFromList(this._evt(event+'$once'));
149 	
150 	function removeFromList(list) {
151 		for (var i=list.length-1; i>=0; i--) {
152 			if (list[i]===listener)
153 				list.splice(i, 1);
154 		}
155 	}
156 };

158 /**
159  * Remove all listeners for a given event.
160  *
161  * @public
162  * @methodOf nanomaps.EventEmitter.prototype
163  * @name removeAllListeners
164  * @param {string} event Event name to add listener to
165  */
166 EventEmitterMethods.removeAllListeners=function(event) {
167 	this._evt(event).length=0;
168 	this._evt(event+'$once').length=0;
169 };

171 /**
172  * Invokes a given event by name, triggering all registered persistent listeners
173  * and once only listeners.  All arguments after the first are taken to be the
174  * arguments to the event listeners.  Once only listeners are invoked first
175  * followed by persistent listeners.  Otherwise, listeners are invoked in the
176  * order added.
177  * <p>
178  * If a method exists on the instance named 'on' + event, then that method
179  * will be invoked as an event listener prior to any others.
180  *
181  * <p><i>TODO: Make listener add/remove durable from within a callback</i>
182  *
183  * @public
184  * @methodOf nanomaps.EventEmitter.prototype
185  * @name emit
186  * @param {string} event Event name to add listener to
187  * @param ... Arguments to registred listeners 
188  */
189 EventEmitterMethods.emit=function(event /*, arg1..argn */) {
190 	var i, list, eventArgs=Array.prototype.slice.call(arguments, 1),
191 		handler=this['on' + event];
192 
193 	// Emit on this object
194 	if (typeof handler==='function') {
195 		handler.apply(this, eventArgs);
196 	}
197 	
198 	// Emit once events
199 	list=this._evt(event+'$once');
200 	for (i=0; i<list.length; i++) {
201 		list[i].apply(this, eventArgs);
202 	}
203 	list.length=0;	// Zero the once only array
204 
205 	// Emit standard events
206 	list=this._evt(event);
207 	for (i=0; i<list.length; i++) {
208 		list[i].apply(this, eventArgs);
209 	}
210 };

212 /**
213  * Overrides the given methodName on this instance applying the given
214  * advice ("before" or "after").  The given target method is invoked
215  * for the advice.
216  * 
217  * <p>Invoking this method supports the arbitrary augmentation of a class's
218  * methods with before or after advice.  The original method is replaced with
219  * a stub that invokes a list of other methods before and after the original
220  * target.  If the original method doesn't exist, the stub will still be
221  * defined.
222  * <p>It is safe to use this method on prototypes or instances as it will only
223  * modify the outermost prototype (target of the call).
224  *
225  * @example MapSurface.prototype.advise('initialize', 'after', 
226  *		function() { alert('Initialized'); });
227  * @example myMap.advise('onzoom', 'after', 
228  *		function() { alert('zoomed'); });
229  * @public
230  * @methodOf nanomaps.EventEmitter.prototype
231  * @name advise
232  * @param {string} methodName Name of the method to override
233  * @param {string} advice Advice to apply ("before" or "after")
234  * @param {function} target Function which is the target of advice
235  */
236 EventEmitterMethods.advise=function(methodName, advice, target) {
237 	var originalMethod=this[methodName],
238 		advisorStub, adviceList;
239 		
240 	if (!originalMethod || !originalMethod.__stub__ || !this.hasOwnProperty(methodName))
241 		advisorStub=this[methodName]=getAdvisorStub(originalMethod);
242 	else
243 		advisorStub=originalMethod;
244 	
245 	adviceList=advisorStub[advice];
246 	if (adviceList) adviceList.push(target);
247 };
248 
249 /**
250  * @private
251  */
252 function getAdvisorStub(method) {
253 	var stub=function() {
254 		var i, thisFunction=arguments.callee,
255 			list, retValue;
256 		
257 		// Apply before advice
258 		list=thisFunction.before;
259 		for (i=0; i<list.length; i++) {
260 			list[i].apply(this, arguments);
261 		}
262 		
263 		// Invoke original method
264 		if (method) retValue=method.apply(this, arguments);
265 		
266 		// Invoke after method
267 		list=thisFunction.after;
268 		for (i=0; i<list.length; i++) {
269 			list[i].apply(this, arguments);
270 		}
271 	};
272 	stub.__stub__=true;
273 	stub.before=[];
274 	stub.after=[];
275 	return stub;
276 }
