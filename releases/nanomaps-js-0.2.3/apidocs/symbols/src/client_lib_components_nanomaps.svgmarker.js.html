1 /**
  2  * nanomaps.svgmarker.js
  3  * Add vector graphics markers to the map.  This source file
  4  * presumes the presence of svg.  If SVG is not present, the
  5  * various classes should not throw exceptions but will not
  6  * display anything.  It is expected that the patterns employed
  7  * here can be used to provide similar graphics for IE browsers
  8  * using VML, if anyone cares.
  9  * <p>
 10  * Unlike ImgMarker instances, which are factories, it is anticipated
 11  * that vector graphics are much less "prototypical" than standard
 12  * image markers.  They also require more state management since
 13  * they are not "simple" map attachments (ie. their display is inherently
 14  * resolution dependent).  As such, each marker in this package wraps
 15  * the physical elements that are attached to the map and can therefore
 16  * only be attached once.  Most manipulation of the elements is left
 17  * to user code acting on the DOM.  This library just takes care of
 18  * physical unit conversions.
 19  */
 20 var svgns='http://www.w3.org/2000/svg';
 21 
 22 function createSvgElement(tagName) {
 23 	try {
 24 		return document.createElementNS(svgns, tagName);
 25 	} catch (e) {
 26 		// Just provide a dummy standin
 27 		var elt=document.createElement('svgdummy');
 28 		if (!elt.style) elt.style={};
 29 	}
 30 }
 31 
 32 /**
 33  * Base class SvgMarker.  Manages an svg canvas and defines the map delegate
 34  * needed to keep it updated.
 35  * @public
 36  * @constructor
 37  */
 38 function SvgMarker(settings) {
 39 	settings=this.settings=settings||{};
 40 	var canvas=createSvgElement('svg');
 41 	canvas.setAttribute('class', settings.className||'');
 42 	canvas.mapeer=this;	// Circular ref ok.  The only browsers who support svg have decent gc
 43 	this.canvas=canvas;	
 44 }
 45 SvgMarker.prototype={
 46 	//// -- mapDelegate methods
 47 	unmanaged: false,
 48 	defaultLayer: 'shadow',
 49 	
 50 	getElement: function(mapSurface) {
 51 		this.owner=mapSurface;
 52 		return this.canvas;
 53 	},
 54 	
 55 	/**
 56 	 * Subclasses should override and must update the size/position
 57 	 * of the canvas and contained elements.
 58 	 */
 59 	mareset: function(map, canvas) {
 60 	},
 61 	
 62 	//// -- stub methods intended for subclassing
 63 	
 64 	//// -- regular instance methods
 65 	
 66 	/**
 67 	 * Resets the element.  This typically happens automatically on map updates
 68 	 * but may need to be triggered manually on property update.
 69 	 * @public
 70 	 */
 71 	invalidate: function() {
 72 		var owner=this.owner, canvas=this.canvas;
 73 		if (owner && canvas.parentNode) 
 74 			this.mareset(owner, canvas);
 75 	}
 76 };
 77 inherits(SvgMarker, EventEmitter);
 78 
 79 function EllipseMarker(settings) {
 80 	SvgMarker.call(this, settings);
 81 	var ellipse=createSvgElement('ellipse');
 82 	this.canvas.appendChild(ellipse);
 83 	this.ellipse=ellipse;
 84 }
 85 EllipseMarker.prototype={
 86 	mareset: function(map, canvas) {
 87 		var ellipse=this.ellipse,
 88 			settings=this.settings,
 89 			rx, ry,
 90 			unit=settings.unit||'px',
 91 			longitude=settings.longitude,
 92 			latitude=settings.latitude,
 93 			canvasWidth, canvasHeight,
 94 			centerX, centerY,
 95 			xy, x, y;
 96 
 97 		// Either calculate rx/ry independently (true ellipse)
 98 		// or just take the radius and fake it to create circles
 99 		// visually.  Accurate for small areas and makes more sense
100 		// visibly
101 		if ('radius' in settings) {
102 			rx=ry=translateX(map, settings.radius||0, unit);
103 		} else {
104 			rx=translateX(map, settings.rx||0, unit);
105 			ry=translateY(map, settings.ry||0, unit);
106 		}
107 			
108 		canvasWidth=oddCeil(rx*2+20);
109 		canvasHeight=oddCeil(ry*2+20);
110 		
111 		centerX=Math.round(canvasWidth/2);
112 		centerY=Math.round(canvasHeight/2);
113 		
114 		ellipse.setAttribute('cx', centerX);
115 		ellipse.setAttribute('cy', centerY);
116 		ellipse.setAttribute('rx', rx);
117 		ellipse.setAttribute('ry', ry);
118 		
119 		//alert('rx=' + rx);
120 		
121 		xy=map.globalToXY(settings);
122 		if (xy) {
123 			x=xy.x()-centerX;
124 			y=xy.y()-centerY;
125 			canvas.style.left=x+'px';
126 			canvas.style.top=y+'px';
127 			canvas.style.width=canvasWidth+'px';
128 			canvas.style.height=canvasHeight+'px';
129 			canvas.style.display='block';
130 		} else {
131 			canvas.style.display='none';
132 		}
133 	}
134 };
135 inherits(EllipseMarker, SvgMarker);

137 //// -- utilities
138 function oddCeil(n) {
139 	n=Math.ceil(n);
140 	if ((n%2)!==1) n++;
141 	return n;
142 }
143 
144 function translateX(map, distance, unit) {
145 	var res=map.mapState.res;
146 	switch (unit) {
147 	case 'px': return distance;
148 	case 'm':
149 		return distance/res;
150 	default:
151 		return 0;
152 	}
153 }
154 
155 function translateY(map, distance, unit) {
156 	var res=map.mapState.res;
157 	switch (unit) {
158 	case 'px': return distance;
159 	case 'm':
160 		return distance/res;
161 	default:
162 		return 0;
163 	}
164 }
