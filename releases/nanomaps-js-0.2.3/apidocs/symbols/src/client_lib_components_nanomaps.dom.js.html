  7 // Feature detection
  8 var hasAddEventListener=!!window.addEventListener;
  9 
 10 function getComputedStyle(elt, prop) {
 11 	if (elt.currentStyle) {
 12 		// IE
 13 		return elt.currentStyle[prop];
 14 	} else if (window.getComputedStyle) {
 15 		// DOM
 16 		return elt.ownerDocument.defaultView.getComputedStyle(elt,'').getPropertyValue(prop);
 17 	}
 18 	return '';
 19 }
 20 
 21 function isPositioned(elt) {
 22 	var attr=getComputedStyle(elt, 'position');
 23 	return attr==='relative' || attr==='absolute' || attr==='fixed';
 24 }
 25 
 26 function isHtmlElement(object) {
 27 	// Detect if HTMLElement in standards compliant way with fallbacks for IE
 28 	return window.HTMLElement ? object instanceof HTMLElement : (object.nodeType===1);
 29 }
 30 
 31 function addEventListener(target, eventName, listener) {
 32 	target.addEventListener(eventName, listener, false);
 33 }
 34 
 35 function removeEventListener(target, eventName, listener) {
 36 	target.removeEventListener(eventName, listener, false);
 37 }
 38 
 39 function isLeftClick(event) {
 40 	// TODO: IE needs some TLC here.  For it, button==1 for
 41 	// left click
 42 	if (hasAddEventListener) {
 43 		return event.button===0;
 44 	} else {
 45 		return event.button===1;
 46 	}
 47 }
 48 
 49 function stopEvent(event) {
 50 	if (event.stopPropagation) {
 51 		event.preventDefault();
 52 		event.stopPropagation();
 53 	} else {
 54 		event.cancelBubble = true;
 55 		event.returnValue = false;
 56 	}
 57 }
 58 
 59 function div() {
 60 	return document.createElement('div');
 61 }
