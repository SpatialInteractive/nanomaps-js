 10 /**
 11  * Construct a standard TileLayer that can be attached to a MapSurface.  If a
 12  * TileSelector is not passed in the options, then a default is constructed and
 13  * options are passed to it.  See the options accepted by the TileSelector
 14  * constructor for additional parameters.
 15  *
 16  * @example 
 17  * var map=new nanomaps.MapSurface(someElement);
 18  * map.attach(new nanomaps.TileLayer({ 
 19  *    tileSrc: "http://otile${modulo:1,2,3}.mqcdn.com/tiles/1.0.0/osm/${level}/${tileX}/${tileY}.png" })); 
 20  *
 21  * @class
 22  * @name nanomaps.TileLayer
 23  * @param {TileSelector} [options.selector] describes the source and geometry of the
 24  * tiles
 25  * @param {integer} [options.buffer=64] The number of pixels to actively buffer on
 26  * @param {Number} [options.pixelRatio=1.0] The ratio of tilepixel / csspixel
 27  * @param {Number} [options.autoPixelRatio=false] If no pixelRatio is given consult the browser to determine native ratio
 28  * all sides of the map
 29  */
 30 function TileLayer(options) {
 31 	if (!options) options={};
 32 	/**
 33 	 * The TileLayerPeer object
 34 	 * @name peer
 35 	 * @public
 36 	 * @memberOf nanomaps.TileLayer#
 37 	 */
 38 	var peer=this.peer=new TileLayerPeer();
 39 	var sel=peer.sel=new CartesianTileSelector(options);
 40 	
 41 	/**
 42 	 * Ratio of tilePixel / cssPixel.  This will be 1.0 on most displays.
 43 	 * On high density displays, setting this to 2.0 may result in better
 44 	 * visual clarity.
 45 	 * @name pixelRatio
 46 	 * @public
 47 	 * @memberOf nanomaps.TileLayer#
 48 	 */
 49 	var explicitPixelRatio=Number(options.pixelRatio);
 50 	if (!explicitPixelRatio && options.autoPixelRatio) {
 51 		explicitPixelRatio=window.devicePixelRatio;
 52 	}
 53 	
 54 	/**
 55 	 * Ratio of tilePixel / cssPixel.  This will be 1.0 on most displays.
 56 	 * On high density displays, setting this to 2.0 may result in better
 57 	 * visual clarity.
 58 	 * @name pixelRatio
 59 	 * @public
 60 	 * @memberOf nanomaps.TileLayer#
 61 	 */
 62 	sel.pixelRatio=peer.pixelRatio=this.pixelRatio=explicitPixelRatio||1.0;
 63 	
 64 	/**
 65 	 * The element that is the root of the TileLayer
 66 	 * @name element
 67 	 * @public
 68 	 * @memberOf nanomaps.TileLayer#
 69 	 */
 70 	var element=this.element=document.createElement('div');
 71 	element.style.position='absolute';
 72 	element.style.left='0px';
 73 	element.style.top='0px';
 74 	if (explicitPixelRatio) {
 75 		element.style.zoom=(1/explicitPixelRatio);
 76 	}
 77 	element.mapeer=peer;
 78 	element.setAttribute('tilesrc', sel.toString());	// For DOM debugging
 79 }
 80 TileLayer.prototype={
 81 	/**
 82 	 * (Attachment Api) Tile layers are unmanaged attachments
 83 	 * @public
 84 	 * @memberOf nanomaps.TileLayer.prototype
 85 	 */
 86 	unmanaged: true,
 87 	
 88 	/**
 89 	 * (Attachment Api) Default attachment layer is 'map'.
 90 	 * For touch support it is imperitive that the map layer
 91 	 * is below the event layer because touch state will be lost
 92 	 * if the event ever attaches to nodes that are removed
 93 	 * are the target of a touch event.
 94 	 * @public
 95 	 * @memberOf nanomaps.TileLayer.prototype
 96 	 */
 97 	defaultLayer: 'map',
 98 
 99 	/**
100 	 * (Attachment Api) Get the element that should be attached
101 	 * to the map.
102 	 * @public
103 	 * @memberOf nanomaps.TileLayer.prototype
104 	 */
105 	getElement: function(mapSurface) {
106 		/**
107 		 * When a TileLayer has been attached to a map, its owner
108 		 * property is the owning MapSurface
109 		 * @public
110 		 * @memberOf nanomaps.TileLayer#
111 		 */
112 		this.owner=mapSurface;
113 		return this.element;
114 	}
115 };

117 /**
118  * Peer object for TileLayers
119  * @private
120  * @name nanomaps.TileLayerPeer
121  * @constructor
122  */
123 function TileLayerPeer() {
124 	// TileSets
125 	this.lockedState=null;
126 	this.current=new TileSet();
127 	this.old=new TileSet();
128 	this.transition=new TileSet();
129 }
130 TileLayerPeer.prototype={
131 	/**
132 	 * @private
133 	 */
134 	maposition: function(map, element) {
135 		this.mareset(map, element);
136 	},
137 	
138 	/**
139 	 * Populates the transition TileSet with tiles for the given mapState.
140 	 * A lot of this code is semi-duplicated in onreset but with minor twists
141 	 * and I've not been able to condense them into one without breaking things.
142 	 * @private
143 	 */
144 	loadPendingTiles: function(mapState) {
145 		var transitionTileSet=this.transition,
146 			currentTileSet=this.current,
147 			pixelRatio=this.pixelRatio,
148 			right=pixelRatio * (mapState.w-1),
149 			bottom=pixelRatio * (mapState.h-1),
150 			updatedKeys,
151 			i,
152 			key,
153 			tile,
154 			newTiles=[];
155 		
156 		//console.log('Loading pending tiles for target state ' + mapState.res);
157 		
158 		transitionTileSet.clear();
159 		// Select tiles that intersect our display area
160 		updatedKeys=this.sel.select(mapState.prj,
161 			mapState.res / pixelRatio,
162 			mapState.getPrjX(0,0),
163 			mapState.getPrjY(0,0),
164 			mapState.getPrjX(right,bottom),
165 			mapState.getPrjY(right,bottom));
166 		
167 		// Match them up against what we are already displaying
168 		for (i=0; i<updatedKeys.length; i++) {
169 			key=updatedKeys[i];
170 			tile=currentTileSet.move(key, transitionTileSet);
171 			if (!tile) {
172 				// We only create a new transitional tile if the tile isn't
173 				// on the current display
174 				tile=new Tile(this.sel, key);
175 				transitionTileSet.add(tile);
176 				newTiles.push(tile);
177 				setTileBounds(mapState, tile, pixelRatio);
178 			}
179 		}
180 		
181 		// Sort and load
182 		sortTiles(newTiles, pixelRatio*mapState.w/2, pixelRatio*mapState.h/2);
183 		for (i=0; i<newTiles.length; i++) {
184 			tile=newTiles[i];
185 			// Load the tiles but don't generate previews because
186 			// no other tilesets are referenced to the same display.
187 			// We'll need to generate previews when we actually go
188 			// to use them later
189 			tile.loading=true;
190 			this.sel.loadTile(tile, null, true);
191 		}
192 	},
193 	
194 	/**
195 	 * TODO: Port transition parts back in later
196 	 * @private
197 	 */
198 	mareset: function(map, element) {
199 		var currentTileSet=this.current,
200 			transitionTileSet=this.transition,
201 			oldTileSet=this.old,
202 			mapState=map.mapState,
203 			lockedState=this.lockedState,
204 			pixelRatio=this.pixelRatio,
205 			right=pixelRatio*(mapState.w-1),
206 			bottom=pixelRatio*(mapState.h-1),
207 			updatedKeys,
208 			i,
209 			key,
210 			tile,
211 			newTiles=[];
212 		
213 		if (lockedState!==mapState.finalState) {
214 			// Handle transition and detransition
215 			lockedState=this.lockedState=mapState.finalState;
216 
217 			// Dump pending tiles into current for consideration in the main loop
218 			//console.log('Transitional end state changed: Sweeping tiles to current');
219 			//transitionTileSet.sweep(currentTileSet);
220 			//console.log('Tiles swept to current');
221 			
222 			if (lockedState) {
223 				// Load tiles for a new final state
224 				this.loadPendingTiles(lockedState);
225 			}
226 		}
227 		
228 		//console.log('TileLayer.onreset(transition locked=' + (!!lockedState) + ')');
229 		currentTileSet.resetMarks();
230 		
231 		// Select tiles that intersect our display area
232 		updatedKeys=this.sel.select(mapState.prj,
233 			mapState.res / pixelRatio,
234 			mapState.getPrjX(0,0),
235 			mapState.getPrjY(0,0),
236 			mapState.getPrjX(right,bottom),
237 			mapState.getPrjY(right,bottom));
238 
239 		// Match them up against what we are already displaying
240 		for (i=0; i<updatedKeys.length; i++) {
241 			key=updatedKeys[i];
242 			tile=currentTileSet.get(key) || transitionTileSet.move(key, currentTileSet);
243 
244 			// If we're not in a transitional state.
245 			// If the tile is marked temporary, then disregard it
246 			// because we don't want half rendered crap anymore.
247 			// Just the good stuff for us from here on out.
248 			if (!lockedState && tile && tile.temporary) tile=null;
249 
250 			// Still no tile?  Create one.
251 			if (!tile) {
252 				tile=new Tile(this.sel, key);
253 				if (lockedState) tile.temporary=true;
254 				currentTileSet.add(tile);
255 			}
256 			tile.mark=true;
257 			
258 			setTileBounds(mapState, tile, pixelRatio);
259 
260 			// After a transition we may be dealing with a child that
261 			// was loaded without ever being attached.  Fix that now.
262 			if (!tile.parent) {
263 				// Not yet added to display.  Put it in the list
264 				// to be fixed up as a new tile
265 				newTiles.push(tile);
266 			}
267 		}
268 		
269 		// If we are generating previews, then we need to sweep
270 		// no longer used tiles into the oldTileSet and update their
271 		// display metrics so that the new tiles can use them for
272 		// previews
273 		if (newTiles.length>0) {
274 			currentTileSet.sweep(oldTileSet, function(tile) {
275 				setTileBounds(mapState, tile, pixelRatio);
276 			});
277 		}
278 		
279 		// newTileRecords now contains all records that have been newly allocated.
280 		// We initialize them here in this ass-backwards way because we want to sort
281 		// them by proximity to the center but don't have the display information until
282 		// after we've iterated over all of them.  Think of this as the "initialize new
283 		// tiles" loop
284 		sortTiles(newTiles, pixelRatio * mapState.w/2, pixelRatio * mapState.h/2);
285 		for (i=0; i<newTiles.length; i++) {
286 			tile=newTiles[i];
287 			
288 			// Order is important here.  The tile may have a drawable
289 			// set within the call to loadTile, in which case don't do
290 			// the work to generate a preview
291 			if (!tile.loading && !tile.temporary) {
292 				tile.loading=true;
293 				this.sel.loadTile(tile);
294 			}
295 			if (!tile.drawable) {
296 				tile.generatePreview(oldTileSet);
297 			}
298 			tile.attach(element);
299 		}
300 		
301 		currentTileSet.sweep();
302 		oldTileSet.sweep();
303 		if (!lockedState) transitionTileSet.clear();
304 	}
305 	
306 };

308 /**
309  * Sort tiles based on their center coordinate's proximity to
310  * some reference coordinate.  The idea is that we are trying to
311  * load tiles near the center before tiles on the edges.
312  */
313 function sortTiles(tilesAry, x, y) {
314 	tilesAry.sort(function(tile1, tile2) {
315 		var score1=Math.abs(tile1.left+tile1.width/2 - x) + Math.abs(tile1.top+tile1.height/2 - y),
316 			score2=Math.abs(tile2.left+tile2.width/2 - x) + Math.abs(tile2.top+tile2.height/2 - y);
317 		return score1 - score2;
318 	});
319 }

321 /**
322  * Given a MapState and TileKey, fill in a Rect with the pixel coordinates
323  * of the tile for the current state.
324  * This assumes rectangular display.  For rotation, this will all need to
325  * be reworked.
326  * @param mapState
327  * @param tile
328  */
329 function setTileBounds(mapState, tile, pixelRatio) {
330 	var size=tile.sel.tileSize,
331 		tileKey=tile.tileKey,
332 		scaledSize=pixelRatio * Math.ceil(size * tileKey.res / mapState.res),
333 		left=pixelRatio * Math.floor(mapState.prjToDspX(tileKey.scaledX * tileKey.res) - mapState.x),
334 		top=pixelRatio * Math.floor(mapState.prjToDspY(tileKey.scaledY * tileKey.res) - mapState.y);
335 		
336 	tile.setBounds(left, top, scaledSize, scaledSize);
337 }

339 /**
340  * Class representing a tile selector for the layout of OSM, MS, Google, et al.
341  * <p>
342  * Typically, all that will need to be specified is options.tileSrc.  This string
343  * will be interpolated relative to the tile information with the following
344  * substitutions made:
345  * <ul>
346  * <li>${level} - Native level of the tile
347  * <li>${tileX} - X coordinate of the tile
348  * <li>${tileY} - Y coordinate of the tile
349  * <li>${modulo:a,b} - Select one of the values in the list (in this case 'a' and 'b')
350  *     based on a stable modulus based on the tile coordinate system.  Used to
351  *     pick a domain name for a given tile.
352  * <li>${quadkey} - Microsoft quadkey representation of tile index
353  * </ul>
354  * <p>
355  * Remember to always provide proper attribution and get permission for using tiles.
356  * <p>
357  * The following tileSrc can be used to display tiles from different providers:
358  * <ul>
359  * <li>OSM (MapQuest): 
360  * 		"http://otile${modulo:1,2,3}.mqcdn.com/tiles/1.0.0/osm/${level}/${tileX}/${tileY}.png"
361  * <li>OSM (Mapnik from openstreetmap.org): "http://${modulo:a,b,c}.tile.openstreetmap.org/${level}/${tileX}/${tileY}.png"
362  * </ul>
363  *
364  * @class
365  * @name nanomaps.CartesianTileSelector
366  * @param {integer} [options.tileSize=256]
367  * @param {string} [options.tileSrc='']
368  */
369 function CartesianTileSelector(options) {
370 	options=options||{};
371 	// set defaults
372 	this.tileSize=options.tileSize||256;
373 	this.srcSpec=options.tileSrc||'';
374 	
375 	// Pending tile records, keyed by tile.tileKey.id:
376 	// {
377 	//		tile: the owning tile
378 	//		src: image source
379 	// 		img: DOM image element
380 	//		loaded: true|false
381 	//		time: utc millis on load, load ms on finish
382 	// }
383 	this.pending={};
384 }
385 CartesianTileSelector.prototype={
386 	toString: function() {
387 		return this.srcSpec;
388 	},
389 	
390 	_sched: function(pended) {
391 		var self=this, 
392 			id=pended.tile.tileKey.id,
393 			img=pended.img;
394 		img.onload=function() {
395 			pended.loaded=true;
396 			finish();
397 			
398 			if (pended.tile) {
399 				pended.tile.update(img);
400 			}
401 		};
402 		img.onerror=function() {
403 			finish();
404 		};
405 		
406 		function finish() {
407 			pended.time=now()-pended.time;
408 			delete self.pending[id];
409 			//console.log('Tile loaded(' + pended.tile.tileKey.id + ') in ' + pended.time + 'ms: ' + pended.src); 
410 		}
411 		
412 		// And away we go
413 		self.pending[id]=pended;
414 		img.src=pended.src;
415 	},
416 	
417 	loadTile: function(tile) {
418 		var pended={
419 			tile: tile,
420 			src: this.resolveSrc(tile.tileKey),// + '?' + now(),
421 			img: document.createElement('img'),
422 			loaded: false,
423 			time: now()
424 		};
425 		
426 		//console.log('Load tile ' + tile.tileKey.level + '@' + tile.tileKey.tileX + ',' + tile.tileKey.tileY);
427 		this._sched(pended);
428 	},
429 
430 	destroyTile: function(tile) {
431 		var id=tile.tileKey.id;
432 		var pended=this.pending[id];
433 		if (pended && pended.tile===tile) {
434 			//console.log('Destroy tile ' + id);
435 			delete this.pending[id];
436 			// This is rumoured to cancel loading the
437 			// image in some browsers
438 			pended.img.src='data:image/png,';
439 		}
440 	},
441 	
442 	destroyDrawable: function(tile, drawable) {
443 		// TODO Not sure we need to do anything here
444 	},
445 	
446 	/**
447 	 * Get an img src from the tileDesc
448 	 * @public
449 	 * @name resolveSrc
450 	 * @methodOf nanomaps.TileSelector.prototype
451 	 * @param {TileKey} TileKey returned from select
452 	 */
453 	resolveSrc: function(tileKey) {
454 		var self=this;
455 		return self.srcSpec.replace(/\$\{([A-Za-z]+)(\:([^\}]*))?\}/g, function(s, name, ignore, args) {
456 			if (name==='quadkey') {
457 				return tileXYToQuadkey(tileKey.tileX, tileKey.tileY, tileKey.level);
458 			} else if (name==='modulo') {
459 				// get the args and modulo it by the tileDesc.tileX
460 				args=args.split(/\,/);
461 				return args[tileKey.tileX%args.length];
462 			} else if (name==='pixelRatio') {
463 				return self.pixelRatio || 1.0;
464 			} else {
465 				return tileKey[name];
466 			}
467 		});
468 	},
469 	
470 	/**
471 	 * Select all tiles that intersect the given bounding box at the
472 	 * given resolution.  Return an array of TileKey elements.  The coordinates
473 	 * define a bounding box in projected units (resolution invariant).
474 	 *
475 	 * @public
476 	 * @name select
477 	 * @methodOf nanomaps.CartesianTileSelector.prototype
478 	 * @param {Projection} projection map projection
479 	 * @param {Number} resolution resolution at which tiles will be rendered
480 	 * @param {Number} x1
481 	 * @param {Number} y1
482 	 * @param {Number} x2
483 	 * @param {Number} y2
484 	 * @return {Array[TileDesc]}
485 	 */
486 	select: function(projection, resolution, x1, y1, x2, y2) {
487 		// - Projection setup
488 		var tileSize=this.tileSize,
489 			projectedBounds=projection.PRJ_EXTENT,
490 			xinversion=projection.XINVERTED,
491 			yinversion=projection.YINVERTED,
492 			nativeLevel=Math.round(projection.toLevel(resolution)),
493 			nativeResolution=projection.fromLevel(nativeLevel),
494 			nativeOriginX,
495 			nativeOriginY;
496 			
497 		x1/=nativeResolution;
498 		y1/=nativeResolution;
499 		x2/=nativeResolution;
500 		y2/=nativeResolution;
501 		
502 		// Axis inversion madness - gotta love it
503 		if (yinversion) {
504 			// Common path
505 			nativeOriginY=projectedBounds.maxy / nativeResolution;
506 			y1=nativeOriginY - y1;
507 			y2=nativeOriginY - y2;
508 		} else {
509 			nativeOriginY=projectedBounds.miny / nativeResolution;
510 			y1=y1 - nativeOriginY;
511 			y2=y2 - nativeOriginY;
512 		}
513 		
514 		if (xinversion) {
515 			nativeOriginX=projectedBounds.maxx / nativeResolution;
516 			x1=nativeOriginX - x1;
517 			x2=nativeOriginX - x2;
518 		} else {
519 			// Common path
520 			nativeOriginX=projectedBounds.minx / nativeResolution;
521 			x1=x1 - nativeOriginX;
522 			x2=x2 - nativeOriginX;
523 		}
524 		
525 		// - Iteration
526 		var startX=Math.floor(Math.min(x1,x2)/tileSize),
527 			startY=Math.floor(Math.min(y1,y2)/tileSize),
528 			endX=Math.floor(Math.max(x1,x2)/tileSize),
529 			endY=Math.floor(Math.max(y2,y2)/tileSize),
530 			i, j,
531 			projectedX, projectedY,
532 			ret=[];
533 	
534 		for (j=startY; j<=endY; j++) {
535 			if (yinversion) {
536 				// Common path
537 				projectedY=nativeOriginY - j*tileSize;
538 			} else {
539 				projectedY=nativeOriginY + j*tileSize;
540 			}
541 				
542 			for (i=startX; i<=endX; i++) {
543 				if (xinversion) {
544 					projectedX=nativeOriginX - i*tileSize;
545 				} else {
546 					// Common path
547 					projectedX=nativeOriginX + i*tileSize;
548 				}
549 				
550 				ret.push(new CartesianTileKey(
551 					nativeLevel,
552 					i,
553 					j,
554 					nativeResolution,
555 					projectedX,
556 					projectedY,
557 					tileSize
558 				));
559 			}
560 		}
561 		
562 		return ret;
563 	}
564 };

566 /**
567  * Uniquely identified a tile from a given TileSelector and contains
568  * discreet vital fields needed to make rendering decisions about a tile.
569  * Upon construction, the id field is computed as a string from the
570  * tuple (level,tileX,tileY) and can be used as an object key.
571  * <p>
572  * The fields level,tileX,tileY uniquely identitfy the tile.  The
573  * fields res, scaledX, scaledY, size are used to make rendering
574  * decisions.
575  * <p>
576  * The choice of using scaledX,scaledY instead of raw units was not
577  * necessarily a good one.  It potentially saves one multiple/divide
578  * on render, but introduces complexity.  It is retained in this port
579  * for consistency.
580  */
581 function CartesianTileKey(level, tileX, tileY, res, scaledX, scaledY, size) {
582 	this.id=tileX+','+tileY+'@'+level;
583 	this.level=level;
584 	this.tileX=tileX;
585 	this.tileY=tileY;
586 	this.res=res;
587 	this.scaledX=scaledX;
588 	this.scaledY=scaledY;
589 	this.size=size;
590 }

592 /**
593  * A renderable tile, identified by tileKey.  In the JavaScript/HTML port,
594  * the "drawable" is always an HTML element (in android, it is a Drawable).
595  * Note that it is not necessarily an img.  It may be a composite such as
596  * a canvas or a div in cases where a preview is generated from multiple
597  * bits of source material.
598  */
599 function Tile(sel, tileKey) {
600 	this.sel=sel;
601 	this.tileKey=tileKey;
602 	this.drawable=null;
603 	this.parent=null;
604 	this.left=null;
605 	this.top=null;
606 	this.width=null;
607 	this.height=null;
608 	this.mark=false;
609 	this.temporary=false;
610 	this.loading=false;
611 }
612 Tile.prototype={
613 	_commitBounds: function(drawable) {
614 		/*
615 		if (this.width!==this.tileKey.size || this.height!==this.tileKey.size) {
616 			console.log('TileSize round error: (' + this.left + ',' + this.top + ') -> (' + this.width + ',' + this.height + ')');
617 		}
618 		*/
619 		
620 		drawable.style.position='absolute';
621 		drawable.style.left=this.left+'px';
622 		drawable.style.top=this.top+'px';
623 		drawable.style.width=this.width+'px';
624 		drawable.style.height=this.height+'px';
625 	},
626 	
627 	/**
628 	 * Sets a new drawable.  If the tile is rendered, then
629 	 * it will be directly updated in the display hierarchy.
630 	 */
631 	update: function(drawable) {
632 		var orig=this.drawable;
633 		this.drawable=drawable;
634 		drawable.setAttribute('tileid', this.tileKey.id);
635 		
636 		// If we're not rendered, just set it
637 		if (this.parent) {
638 			// Otherwise, do some gyrations
639 			this._commitBounds(drawable);
640 			if (orig) {
641 				this.parent.replaceChild(drawable, orig);
642 			} else {
643 				this.parent.appendChild(drawable);
644 			}
645 		}
646 		
647 		// Get rid of old
648 		if (orig) {
649 			this.sel.destroyDrawable(this, orig);
650 		}
651 	},
652 	
653 	setBounds: function(left, top, width, height) {
654 		var drawable=this.drawable;
655 		this.left=parseInt(left);
656 		this.top=parseInt(top);
657 		this.width=parseInt(width);
658 		this.height=parseInt(height);
659 		if (drawable) this._commitBounds(drawable);
660 	},
661 	
662 	/**
663 	 * Attach the tile to the display
664 	 */
665 	attach: function(parent) {
666 		var drawable=this.drawable;
667 		
668 		if (this.parent && parent!==this.parent) {
669 			// Reparent (doesn't happen but being complete)
670 			this.detach();
671 		}
672 		this.parent=parent;
673 		
674 		if (drawable) {
675 			this._commitBounds(drawable);
676 			if (!drawable.parentNode) parent.appendChild(drawable);
677 		}
678 	},
679 	
680 	/**
681 	 * Remove a tile from the view
682 	 */
683 	detach: function() {
684 		var drawable=this.drawable;
685 		this.parent=null;
686 		if (drawable&&drawable.parentNode) {
687 			drawable.parentNode.removeChild(drawable);
688 		}
689 	},
690 	
691 	dispose: function() {
692 		this.detach();
693 		if (this.drawable) {
694 			this.sel.destroyDrawable(this, this.drawable);
695 			this.drawable=null;
696 		}
697 		this.sel.destroyTile(this);
698 	},
699 	
700 	/**
701 	 * Generate a preview Drawable for this tile from material in the
702 	 * source TileSet.  This assumes that this tile's bounds are updated
703 	 * to current display bounds and all tiles in the tileSet are also
704 	 * updated.  If a preview could be generated it's drawable will be
705 	 * set
706 	 */
707 	generatePreview: function(tileSet) {
708 		var self=this,
709 			preview=null;
710 		
711 		tileSet.intersect(self, function(srcTile) {
712 			if (!preview) {
713 				try {
714 					preview=new TileCompositor(self);
715 				} catch (e) {
716 					// Will throw an exception if the browser
717 					// doesn't support canvas.  Oh well.
718 					// Uncomment to see reall error.
719 				}
720 			}
721 			
722 			if (preview) preview.add(srcTile);
723 		});
724 		
725 		if (preview) {
726 			self.update(preview.drawable);
727 		}
728 	}
729 };

731 function TileCompositor(tile) {
732 	var canvas=document.createElement('canvas'),
733 		ctx;
734 	canvas.width=tile.width;
735 	canvas.height=tile.height;
736 	ctx=canvas.getContext('2d');
737 	
738 	// Uncomment to see the drawn boxes
739 	//ctx.fillStyle='rgb(200,0,0)';
740 	//ctx.fillRect(2,2,canvas.width-2,canvas.height-2);
741 	
742 	// - exports
743 	this.drawable=canvas;
744 	this.add=function(blitTile) {
745 		var dx=blitTile.left-tile.left, dy=blitTile.top-tile.top,
746 			sx=0, sy=0,
747 			sw, sh,
748 			overflow,
749 			w=blitTile.width,
750 			h=blitTile.height,
751 			blitScaleX=blitTile.tileKey.size / blitTile.width, 
752 			blitScaleY=blitTile.tileKey.size / blitTile.height;
753 		
754 		if (dx<0) {
755 			// Clip left
756 			sx=-dx;
757 			dx=0;
758 			w-=sx;
759 		}
760 		if (dy<0) {
761 			// Clip top
762 			sy=-dy;
763 			dy=0;
764 			h-=sy;
765 		}
766 		
767 		// Clip right
768 		overflow=dx+w-tile.width;
769 		if (overflow>=0) {
770 			w-=overflow;
771 		}
772 		
773 		// Clip bottom
774 		overflow=dy+h-tile.height;
775 		if (overflow>=0) {
776 			h-=overflow;
777 		}
778 		
779 		// Source coordinates have been figured according to the current display
780 		// but the drawImage call works on original image dimensions.
781 		// Therefore, scale the source coordinates
782 		sx*=blitScaleX;
783 		sy*=blitScaleY;
784 		sw=w*blitScaleX;
785 		sh=h*blitScaleY;
786 		
787 		//console.log('drawImage(' + sx + ',' + sy + ',' + sw + ',' + sh + ',' + dx + ',' + dy + ',' + w + ',' + h + ')');
788 		try {
789 			ctx.drawImage(blitTile.drawable,
790 				sx, sy,
791 				sw, sh,
792 				dx, dy,
793 				w, h);
794 		} catch (e) {
795 			// There are boundary conditions due to rounding where
796 			// some coordinates may be out of bounds.  These will be
797 			// for tiles just outside of the bounds and we're just
798 			// going to let the system catch it and not lose any
799 			// sleep about it
800 			//console.log('Exception in call to drawImage(' + sx + ',' + sy + ',' + sw + ',' + sh + ',' + dx + ',' + dy + ',' + w + ',' + h + ')');
801 		}
802 	};
803 }

805 /**
806  * Maintains a set of Tile instances that are locked in some way.
807  * The rendering process typically consists of visiting tiles in a set,
808  * marking as it goes and then sweeping unmarked tiles.
809  */
810 function TileSet() {
811 	/**
812 	 * Content object keyed by TileKey.id with values of Tile
813 	 */
814 	this._c={};
815 }
816 TileSet.prototype={
817 	each: function(callback) {
818 		var c=this._c, k;
819 		for (k in c) {
820 			if (c.hasOwnProperty(k)) {
821 				callback(c[k]);
822 			}
823 		}
824 	},
825 	
826 	/**
827 	 * Reset all marks on contained tiles to false
828 	 */
829 	resetMarks: function() {
830 		this.each(function(tile) {
831 			tile.mark=false;
832 		});
833 	},
834 	
835 	get: function(tileKey) {
836 		return this._c[tileKey.id];
837 	},
838 	
839 	add: function(tile) {
840 		var id=tile.tileKey.id, c=this._c;
841 		var existing=c[id];
842 		if (existing) {
843 			existing.dispose();
844 		}
845 		c[id]=tile;
846 	},
847 	
848 	move: function(tileKey, destTileSet) {
849 		var id=tileKey.id, c=this._c,
850 			existing=c[id];
851 		if (existing) {
852 			delete c[id];
853 			destTileSet.add(existing);
854 		}
855 		return existing;
856 	},
857 	
858 	/**
859 	 * Sweep unmarked tiles into the targetTileSet,
860 	 * optionally invoking tileCallback(tile) for each
861 	 * tile transferred
862 	 */
863 	sweep: function(targetTileSet, tileCallback) {
864 		var c=this._c, ary=[], k, tile;
865 		for (k in c) {
866 			if (c.hasOwnProperty(k)) {
867 				tile=c[k];
868 				if (!tile.mark) {
869 					if (targetTileSet) {
870 						targetTileSet.add(tile);
871 					} else {
872 						tile.dispose();
873 					}
874 					if (tileCallback) tileCallback(tile);
875 					ary.push(k);
876 				}
877 			}
878 		}
879 		
880 		for (k=0; k<ary.length; k++) {
881 			delete c[ary[k]];
882 		}
883 	},
884 	
885 	/**
886 	 * Dispose of all tiles and clear
887 	 */
888 	clear: function() {
889 		var c=this._c, ary=[], k, tile;
890 		for (k in c) {
891 			if (c.hasOwnProperty(k)) {
892 				tile=c[k];
893 				tile.dispose();
894 			}
895 		}
896 		this._c={};
897 	},
898 	
899 	/**
900 	 * Select all tiles whose bounds intersect the bounds of the
901 	 * given tile and have a drawable.
902 	 */
903 	intersect: function(srcTile, callback) {
904 		var c=this._c, k, tile,
905 			minx=srcTile.left, miny=srcTile.top,
906 			maxx=minx+srcTile.width, maxy=miny+srcTile.height;
907 		for (k in c) {
908 			if (c.hasOwnProperty(k)) {
909 				tile=c[k];
910 				if (tile.drawable && !(
911 						tile.left>=maxx ||
912 						(tile.left+tile.width)<minx ||
913 						tile.top>=maxy ||
914 						(tile.top+tile.height)<miny
915 					))
916 				{
917 					// Intersects
918 					callback(tile);
919 				}
920 			}			
921 		}
922 	}
923 };
