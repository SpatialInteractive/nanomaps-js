  6 /**
  7  * The entire viewable state of the map is represented by the
  8  * following vital parameters:
  9  * <ul>
 10  * <li>prj - Map projection
 11  * <li>res - resolution (projected units/pixel)
 12  * <li>x - origin of the upper left corner of the map
 13  * <li>y - origin of the upper left corner of the map
 14  * <li>w - width of the map area
 15  * <li>h - height of the map area
 16  * </ul>
 17  * 
 18  * Representing this as a value object aids when dealing
 19  * with transitions, animations, etc where we want to interpolate
 20  * between two or more MapState instances.
 21  * <p>
 22  * The MapSurface holds the "master" MapState in its mapState
 23  * property.  This property can be read but should only be modified
 24  * through appropriate API calls.
 25  * <p>
 26  * Note that the MapState is a quasi-internal object.  It is used directly
 27  * by code integrating into nanomaps, but users generally will not/should not
 28  * encounter it.  It's methods are terse and no validity checks are done
 29  * on arguments.  That is left to higher level code.
 30  *
 31  * @constructor
 32  * @name nanomaps.MapState
 33  */
 34 function MapState(other) {
 35 	if (other) {
 36 		other.copy(this);
 37 	} else {
 38 		this.prj=null;
 39 		this.res=1;
 40 		this.x=0;
 41 		this.y=0;
 42 		this.w=0;
 43 		this.h=0;
 44 	}
 45 	
 46 	/**
 47 	 * If this state represents a transitional state on the way to
 48 	 * some final state, then that final state will be set here.
 49 	 * This is intended to be used by components that either may just
 50 	 * take a swag during transitions or may wish to preload resources
 51 	 * based on final states instead of intermediate.
 52 	 * @public
 53 	 * @name finalState {MapState}
 54 	 * @memberOf nanomaps.MapState#
 55 	 */
 56 	this.finalState=null;
 57 }

 59 	/**
 60 	 * Detect the level of change between this mapState and a comparison
 61 	 * mapState.
 62 	 * @public
 63 	 * @name compare
 64 	 * @methodOf nanomaps.MapState.prototype
 65 	 * @param other {MapState}
 66 	 * @return {Number} 0 if no change, 1 if position change, 2 if full change
 67 	 */
 68 	compare: function(other) {
 69 		var ret=0;
 70 		if (this.x!==other.x || this.y!==other.y || this.w!==other.w || this.h!==other.h) {
 71 			ret=1;
 72 		}
 73 		
 74 		if (this.prj!==other.prj || this.res!==other.res) {
 75 			ret=2;
 76 		}
 77 		
 78 		return ret;
 79 	},

274 /**
275  * Initialize an existing DOM element as a map.  All internal map structures
276  * are added before any existing content in the element.
277  *
278  * <h2>Sizing</h2>
279  * The map size must be explicitly maintained.  If not specified, then the natural
280  * size of the containing element is used.  If this natural size ever changes,
281  * setSize() must be called to reset it.
282  * 
283  * @example
284  * var map=new nanomaps.MapSurface(someElement);
285  * map.attach(new nanomaps.TileLayer({ 
286  *    tileSrc: "http://otile${modulo:1,2,3}.mqcdn.com/tiles/1.0.0/osm/${level}/${tileX}/${tileY}.png" }));
287  * 
288  * @constructor
289  * @extends nanomaps.EventEmitter
290  * @name nanomaps.MapSurface
291  * @param {HTMLElement} elt The map container element
292  * @param {integer} [options.width] Explicit width of the map
293  * @param {integer} [options.height] Explicit height of the map
294  * @param {number} [options.resolution] Initial resolution
295  * @param {Projection} [options.projection=new nanomaps.WebMercatorProjection()] Map projection 
296  * @param {LatLngObject} [options.center] Initial center of the map
297  */
298 function MapSurface(elt, options) {
299 	if (!options) options={};
300 	var document=options.document||elt.ownerDocument||window.document,
301 		width=options.width, height=options.height, attr,
302 		eventLayer,
303 		mapState,
304 		projection;
305 		
306 	// Local functions
307 	function createElement(name) {
308 		return document.createElement(name);
309 	}
310 	
311 	/**
312 	 * Create an element in the document that this map exists in.
313 	 * @methodOf nanomaps.MapSurface#
314 	 * @name createElement
315 	 * @return {HTMLElement}
316 	 */
317 	this.createElement=createElement;
318 	this._layers=[];
319 	this._surfaces=[];
320 	
321 	// Hardcode some important styles
322 	elt.nmt='parent';
323 	elt.style.overflow='hidden';
324 	if (!isPositioned(elt))
325 		elt.style.position='relative';	// Make positioned
326 	
327 	// Dictionary of elements
328 	this.elements={
329 		document: document,
330 		parent: elt
331 	};
332 	
333 	// Setup the event layer
334 	eventLayer=this.layer(LAYER_NAMES.EVENT);
335 	eventLayer.style.position='absolute';
336 	eventLayer.style.left='0px';
337 	eventLayer.style.top='0px';
338 	eventLayer.style.width='100%';
339 	eventLayer.style.height='100%';
340 	this.elements.event=eventLayer;
341 	
342 	// Display mapState
343 	/**
344 	 * MapState object currently being displayed.  This does not
345 	 * contain uncommitted changes.
346 	 * @public
347 	 * @memberOf nanomaps.MapSurface#
348 	 * @name mapState
349 	 */
350 	this.mapState=mapState=new MapState();
351 	mapState.prj=options.projection||new Projections.WebMercator();
352 	mapState.res=options.resolution||mapState.prj.DEFAULT_RESOLUTION;
353 	this._zoomBias=options.zoomBias||0;
354 	
355 	// Pending mapState
356 	this._pendMapState=new MapState(mapState);
357 	this._pendLock=0;
358 	this._pendAnim=null;
359 	
360 	// Fixed size or autosize
361 	if (typeof width!=='number' || typeof height!=='number') {
362 		this.setSize();
363 		// Auto width/height (get inner size)
364 		width=elt.clientWidth;
365 		height=elt.clientHeight;
366 	} else {
367 		this.setSize(width, height);
368 	}
369 	
370 	// Initialization hook
371 	this.initialize(options);
372 	
373 	// Collect loose content
374 	this.collect();
375 }

392 /**
393  * Clamp the given zoom level to the valid range
394  */
395 function clampZoom(map, level) {
396 	var prj=map.mapState.prj;
397 	level=Number(level);
398 	
399 	// Not much to do here - but letting an NaN in is like inviting
400 	// a vampire into your house
401 	if (isNaN(level)) level=prj.MAX_LEVEL;
402 	
403 	if (level<prj.MIN_LEVEL) level=prj.MIN_LEVEL;
404 	else if (level>prj.MAX_LEVEL) level=prj.MAX_LEVEL;
405 	return level;
406 }

408 var LAYER_NAMES={
409 	BACKGROUND: 0,
410 	MAP: 10,
411 	SHADOW: 50,
412 	EVENT: 100,
413 	OVERLAY: 1000,
414 	FOREGROUND: 10000
415 };

417 function layerIndexToOrdinal(index) {
418 	// See if the index is really a layer name
419 	var ordinal=LAYER_NAMES[String(index).toUpperCase()];
420 	if (ordinal === undefined) {
421 		// Parse it as a number
422 		ordinal=parseInt(index);
423 		if (isNaN(ordinal)) ordinal=LAYER_NAMES.FOREGROUND;
424 	}
425 	return ordinal;
426 }

428 function makeMapStateFramer(map, initialMapState, finalMapState) {
429 	// Copy mapStates (the references we are given are "live")
430 	// and record strides
431 	var resStride=finalMapState.res - initialMapState.res,
432 		xInitial=initialMapState.getPrjX(0,0),
433 		yInitial=initialMapState.getPrjY(0,0),
434 		xStride=finalMapState.getPrjX(0,0) - xInitial,
435 		yStride=finalMapState.getPrjY(0,0) - yInitial,
436 		wStride=finalMapState.w - initialMapState.w,
437 		hStride=finalMapState.h - initialMapState.h;
438 	initialMapState=new MapState(initialMapState);
439 	finalMapState=new MapState(finalMapState);
440 	updateMapState=new MapState(initialMapState);
441 	
442 	//console.log('Animation framer: resStride=' + resStride + ', xStride=' + xStride + ', yStride=' + yStride + ', wStride=' + wStride + ', hStride=' + hStride);
443 	
444 	return function(n, xy, isFinal) {
445 		var pct=xy[0], changeLevel;
446 		if (isFinal) {
447 			updateMapState=finalMapState;
448 			map._pendAnim=null;
449 			map.mapState.finalState=null;
450 		} else {
451 			//console.log('Anim frame ' + pct);
452 			updateMapState.res=initialMapState.res + pct * resStride;
453 			updateMapState.setPrjXY(xInitial + pct * xStride, yInitial + pct * yStride, 0, 0);
454 			updateMapState.w=initialMapState.w + pct * wStride;
455 			updateMapState.h=initialMapState.h + pct * hStride;
456 		}
457 		
458 		changeLevel=updateMapState.compare(map.mapState);
459 		if (changeLevel>0) {
460 			updateMapState.copy(map.mapState);
461 			map._invalidate(changeLevel>1);
462 		}
463 	};
464 }

466 /**
467  * Begin a series of mapState changes including changes to position, size,
468  * resolution and projection.  Calling this method increments an internal
469  * lock count.  If the lock count is zero (default), then all changes to
470  * map state take effect immediately.  If it is >0 (as is the case after a
471  * call to begin), then the changes are "batched" up until a later call
472  * to commit or rollback.
473  *
474  * @public
475  * @methodOf nanomaps.MapSurface.prototype
476  * @name begin
477  * @return true if this starts a new transaction and false if it just incremented
478  * the lock count on an already started transaction
479  */
480 MapSurfaceMethods.begin=function() {
481 	return this._pendLock++ === 0;
482 };
483 
484 /**
485  * Rolls back all pending map state changes made since the first begin().  Note that
486  * nested rollbacks (ie. savepoints in SQL parlance) are not supported.  Rollback 
487  * resets the entire transaction.  Future calls to commit will have no effect since
488  * the transaction lock will be zero.
489  *
490  * @public
491  * @methodOf nanomaps.MapSurface.prototype
492  * @name rollback
493  */
494 MapSurfaceMethods.rollback=function() {
495 	// Copy back from the display mapState
496 	this.mapState.copy(this._pendMapState);
497 	this._pendLock=0;
498 };

500 /**
501  * Commit a pending transaction, optionally animating between the current display
502  * state and changes introduced in the transaction.  Passing any true value as the
503  * animate parameter will enable animation.  If passing a non-null object, then it
504  * will be taken as animation parameters and will be passed through unchanged to the
505  * Animation() constructor.  The following animation options are settable:
506  * <ul>
507  * <li>curve: An animation curve function such as one produced by MakeBezierCurve(x1,y1,x2,y2).
508  * Defaults to a bezier curve with a pronounced tail acceleration
509  * <li>duration: Duration in seconds.  Default to 0.5.
510  * <li>rate: Frame rate in frames/s.  Default=20 but may be automatically adjusted based on platform 
511  * or previous preformance.
512  * </ul>
513  * @public
514  * @methodOf nanomaps.MapSurface.prototype
515  * @name commit
516  * @param animate {boolean|Object} false to disable animation, true or animation options to animate
517  * @return true if this call to commit finished the transaction.  false if this commit
518  * did nothing
519  */
520 MapSurfaceMethods.commit=function(animate) {
521 	if (this._pendLock<=0 || --this._pendLock>0) return false;
522 	
523 	// Let's do it
524 	var pendMapState=this._pendMapState,
525 		mapState=this.mapState,
526 		animOptions;
527 		
528 	var changeLevel=pendMapState.compare(mapState);
529 	if (changeLevel>0) {
530 		// Something changed
531 		// Interrupt any pending animation
532 		if (this._pendAnim) {
533 			this._pendAnim.interrupt();
534 			this._pendAnim=null;
535 			mapState.finalState=null;
536 		}
537 		
538 		if (!animate) {
539 			// Directly update
540 			pendMapState.copy(mapState);
541 			this._invalidate(changeLevel>1);
542 		} else {
543 			// Start an animation
544 			if (typeof animate==='object') animOptions=animate;
545 			mapState.finalState=new MapState(pendMapState);
546 			this._pendAnim=new Animation(makeMapStateFramer(this, mapState, pendMapState), animOptions);
547 			this._pendAnim.start();
548 		}
549 	}
550 	
551 	return true;
552 };

567 /**
568  * All map objects exist within a layer.  Layers are ordered by an ordinal index
569  * and correspond to a div within the main map div.  Each layer div is identified
570  * by an attribute "layer" with its numeric layer value.  In this way, loose objects
571  * that need to exist in the layer hierarchy can have their "layer" attribute
572  * set to have them positioned in the correct place.  Loose objects without a layer
573  * are assumed to have a layer index of infinity, making them foreground objects.
574  * <p>
575  * This method will return a div for the layer.  If the layer does not exist, it
576  * will be created.
577  * <p>
578  * Layers do not establish a positioning context, inheriting the context of the
579  * containing map div.  They also have zero width and height but are set to display
580  * overflow.  The system may make the special events layer positioned with full width
581  * and height, but this is considered a special case.
582  * <p>
583  * The following layer names are acceptable arguments to index in addition to numeric
584  * values:
585  * <ul>
586  * <li>background (0): The background layer will contain background drawables, generally the
587  * thatch pattern that is below the map
588  * <li>map (10): The map layer should contain views for drawing the map (ie MapTileView)
589  * stacked in order of display.
590  * <li>event (100): Touch events are handled from this layer.  This layer is special.
591  * <li>shadow (500): By convention, the SHADOW layer is where you will want to put overlay shadows
592  * <li>overlay (1000): By convention, the OVERLAY layer is where map overlays and markers should
593  * be added
594  * <li>foreground (10000): Extreme foreground layer.  This is where unattached map objects go and
595  * is synonymous with infinity.
596  * </ul>
597  * Layer names are case insensitive.  The event layer is a special 100% width/height layer
598  * that catches all background events.  Anything below this ordinal will not receive any ui
599  * events. 
600  *
601  * @public
602  * @methodOf nanomaps.MapSurface.prototype
603  * @name layer
604  * @param index The layer name or ordinal value of the desired layer
605  * @return layer div corresponding to index
606  */
607 MapSurfaceMethods.layer=function(index) {
608 	var parent=this.elements.parent,
609 		layers=this._layers,
610 		layer, i,
611 		ordinal=layerIndexToOrdinal(index),
612 		childOrdinal,
613 		insertionPoint;
614 		
615 	// Look it up in our cache
616 	for (i=0; i<layers.length; i++) {
617 		layer=layers[i];
618 		if (layer.ordinal===ordinal) return layer;
619 	}
620 	
621 	// Need to create it
622 	layer=this.createElement('div');
623 	layer.ordinal=ordinal;
624 	layer.nmt='layer';
625 	layer.setAttribute('layer', String(ordinal));	// For debugging in inspector.  Not used otherwise.
626 	layer.style.width='0px';
627 	layer.style.height='0px';
628 	layers.push(layer);
629 	
630 	// Find insertion point
631 	for (insertionPoint=parent.firstChild; insertionPoint; insertionPoint=insertionPoint.nextSibling) {
632 		if (insertionPoint.nodeType===1 && insertionPoint.nmt) {
633 			childOrdinal=insertionPoint.ordinal;
634 			if (childOrdinal!==undefined && childOrdinal>ordinal)
635 				break;
636 		}
637 	}
638 	
639 	// Put it before the insertionPoint or as the last element
640 	parent.insertBefore(layer, insertionPoint);
641 	
642 	return layer;
643 };

645 /**
646  * Within each layer is a so-called "managed surface".  This div is maintained as the
647  * first child of each layer and has its position managed by the system in such a way
648  * that children placed within it at coordinates returned by globalXY (TODO) will always
649  * move with the map.
650  * <p>
651  * This method returns the managed surface div associated with a layer.  Most map objects
652  * end up in the managed div.
653  * @public
654  * @methodOf nanomaps.MapSurface.prototype
655  * @name layer
656  * @param index The layer name or ordinal value of the desired layer
657  * @return surface div corresponding to index

942 /**
943  * Adds content to the map.  This is the primary method of attaching DOM content
944  * to the map after construction.  Give it either an HTMLElement or an Attachment
945  * object that supports the Attachable protocol.
946  *
947  * <h2>Design Pattern</h2>
948  * Internally, the MapSurface works solely with content that is raw DOM elements.
949  * It is expected that any active DOM elements on the map can be associated with
950  * an AttachmentPeer object that manages the display lifecycle of the elements
951  * while on the map surface.  Connecting an HTMLElement to its AttachmentPeer can
952  * be done by setting the element's "mapeer" (Map-Attachment Peer) property to its
953  * peer object.  If this property is not set, then each element will share a single
954  * global AttachmentPeer providing default behaviour.  See Default Behavior below.
955  * <p>
956  * It is often desirable to maintain a high-level JavaScript object providing a
957  * structured API for manipulating the element's content.  This API is most easily
958  * expressed on the Attachment object, which can maintain a reference to the MapSurface
959  * and the AttachmentPeer.  This is precisely how the high level provided attachments
960  * function (markers, SVG, TileLayer, etc).
961  * <p>
962  * It is acknowledged that this peer structure is somewhat awkward but it is done to
963  * support reference models that do not require circular references ensnaring DOM references.
964  * In 2011, the state of such things is better than it was designing an API that requires
965  * circular references is still a recipe for creating sluggish garbage collection at best
966  * and browser-bug induced leaks at worst.
967  *
968  * <h2>Objects</h2>
969  * The following classes or interfaces participate to manage an element's display on the map:
970  * <ul>
971  * <li>HTMLElement - Actual DOM element representing the content
972  * <li>Attachment - Optional object that references the element and provides a user-level api
973  * for manipulating the content
974  * <li>AttachmentPeer - Object attached to the HTMLElement which manages the display state
975  * <li>MapSurface - The target of the attachment
976  * <li>DefaultAttachmentPeer - Default peer used if an HTMLElement does not supply its own peer.
977  * </ul>
978  * 
979  * There are multiple ways to combine and/or collapse these items.  An API could be constructed
980  * for example that just adds methods directly to an HTMLElement and makes it do everything.
981  * The built-in map attachments, however follow a pattern with references in the following pattern:
982  * <pre>
983  * 	MapSurface  ----> HTMLElement  ----> AttachmentPeer
984  * 	    ^              ^
985  * 	    |             /
986  * 	Attachment  -----/
987  * </pre>
988  *
989  * An Attachment will host the user-level API and maintain references to the MapSurface and
990  * the HTMLElement.  The HTMLElement references the AttachmentPeer via its mapeer property.
991  * The AttachmentPeer does not maintain a reference to any of the other objects.  All of its
992  * methods accept the objects it needs to operate against as arguments.  In this way, the peer
993  * can either be a stateless singleton or allocated per attachment.
994  * 
995  * <h3>Associating an HTMLElement with an AttachmentPeer</h3>
996  * The peer for an HTMLElement is located in the following way:
997  * <ol>
998  * <li>If the HTMLElement.mapeer property is set, use that
999  * <li>If the HTMLElement.mapself property === true, then the element should
1000  * be considered its own peer
1001  * <li>The global DefaultAttachmentPeer is used
1002  * </ol>
1003  *
1004  * <h3>Attachment API</h3>
1005  * In order for an arbitrary object to be attached to the map, it should support
1006  * the following properties/methods:
1007  * <ul>
1008  * <li>defaultLayer:Number|String - If specified, this is the default layer
1009  * that should host the object (unless if an explicit layer was passed to attach).
1010  * Defaults to "overlay" if not present.
1011  * <li>unmanaged:Boolean - If true, then the attachment is added to the dom
1012  * under the root layer, otherwise, it is added under the surface layer (which
1013  * is positioned based on map movement)
1014  * <li>getElement(MapSurface):HTMLElement - Attaches or reattaches this Attachment
1015  * object to the given MapSurface.  Should return the HTMLElement representing
1016  * the attachment
1017  * </ul>
1018  * 
1019  * Note in particular that there is no detach method.  Attachments can be
1020  * removed by either removing the corresponding HTMLElement or calling MapSurface.detach
1021  * with the Attachment object. 
1022  *
1023  * <h3>AttachmentPeer API</h3>
1024  * Attachment peers should implement the following:
1025  * <ul>
1026  * <li>maposition(MapSurface,HTMLElement) - If defined, called whenever the map's position
1027  * changes
1028  * <li>mareset(MapSurface,HTMLElement) - If defined, called whenever the map's resolution or
1029  * other parameter changes which is considered a "heavy weight" change
1030  * </ul>
1031  *
1032  * <h1>The following is old docs</h1>
1033  * <h2>Display Management</h2>
1034  * The primary thing that map content needs to be able to do is position itself
1035  * relative to a global coordinate system.  Each element managed by the map
1036  * viewport is checked for a special "mapDelegate" property.  If defined, then
1037  * this delegate object provides callbacks that will be invoked on map geometry
1038  * changes and are responsible for displaying the element properly.  See the
1039  * documentation for a complete treatment of MapDelegate objects.
1040  *
1041  * <h2>Default Behavior</h2>
1042  * If an element added to the map does not have a mapDelegate property, the
1043  * default delegate is used instead.  This default delegate defines standard
1044  * and intuitive behavior for positioning a simple object.  The position and
1045  * pixel offset are determined by taking the first defined result from the
1046  * following sequence:
1047  * <ol>
1048  * <li>If the "geo" property exists on the DOM object.  Take the "latitude", 
1049  *		"longitude", "x" and "y" properties.
1050  * <li>Read the "latitude" and "longitude" attributes of the element.
1051  * </ol>
1052  *
1053  * @public
1054  * @methodOf nanomaps.MapSurface.prototype
1055  * @name attach
1056  * @param {HTMLElement or factory} element Element or factory to add
1057  * @param {String or Number} options.layer The target layer
1058  * @return {HTMLElement} the element attached
1059  */
1060 MapSurfaceMethods.attach=function(attachment, options) {
1061 	if (!options) options={};
1062 	var layerSpec=options.layer,
1063 		unmanaged=options.unmanaged,
1064 		parent,
1065 		element=getAttachmentElement(this, attachment);
1066 	
1067 	if (attachment!==element) {
1068 		// Process as attachment
1069 		if (!layerSpec) layerSpec=attachment.defaultLayer;
1070 		unmanaged=attachment.unmanaged;
1071 	}
1072 	
1073 	// Make sure it is positioned
1074 	element.style.position='absolute';
1075 	
1076 	parent=unmanaged ? this.layer(layerSpec) : this.surface(layerSpec);
1077 	if (element.parentNode===parent) {
1078 		// Already attached to the right place
1079 		return;
1080 	}
1081 	
1082 	try {
1083 		parent.appendChild(element);
1084 	} catch (e) {
1085 		throwNotDomElement(element);
1086 	}
1087 	
1088 	// Send it a reset event
1089 	this._notifyResetSingle(element);
1090 	
1091 	return element;
1092 };
