1 /**
  2  * nanomaps.projections.js
  3  * Currently just the WebMercator projection
  4  */
  5  
  6 /**
  7  * @namespace Projections
  8  * @name nanomaps.Projections
  9  */
 10 var Projections={
 11 	/**
 12 	 * Standard "Web Mercator" projection as used by Google, Microsoft, OSM, et al.
 13 	 * Instantiate as new WebMercator(options). 
 14 	 * @class
 15 	 * @name nanomaps.Projections.WebMercator
 16 	 * @param {Number} [options.MIN_LEVEL=1]
 17 	 * @param {Number} [options.MAX_LEVEL=18]
 18 	 */
 19 	WebMercator: function(options) {
 20 		if (!options) options={};
 21 		/**
 22 		 * Default center for this projection
 23 		 * @public
 24 		 * @name DEFAULT_CENTER
 25 		 * @memberOf nanomaps.Projections.WebMercator#
 26 		 */
 27 		this.DEFAULT_CENTER={lat:39.7406, lng:-104.985441};
 28 		/**
 29 		 * Default resolution for this projection
 30 		 * @public
 31 		 * @name DEFAULT_RESOLUTION
 32 		 * @memberOf nanomaps.Projections.WebMercator#
 33 		 */
 34 		this.DEFAULT_RESOLUTION=611.4962;
 35 		
 36 		/**
 37 		 * WebMercator does not have an x inversion
 38 		 * @public
 39 		 * @name XINVERTED
 40 		 * @memberOf nanomaps.Projections.WebMercator#
 41 		 */
 42 		this.XINVERTED=false;
 43 
 44 		/**
 45 		 * WebMercator <b>does</b> have a y inversion
 46 		 * @public
 47 		 * @name XINVERTED
 48 		 * @memberOf nanomaps.Projections.WebMercator#
 49 		 */
 50 		this.YINVERTED=true;
 51 		
 52 		var EARTH_RADIUS=6378137.0,
 53 			DEG_TO_RAD=.0174532925199432958,
 54 			RAD_TO_DEG=57.29577951308232,
 55 			FOURTHPI=0.78539816339744833,
 56 			HALFPI=1.5707963267948966,
 57 			HIGHEST_RES=78271.5170;
 58 		
 59 		/**
 60 		 * Minimum valid level
 61 		 * @public
 62 		 * @name MIN_LEVEL
 63 		 * @memberOf nanomaps.Projections.WebMercator#
 64 		 */
 65 		this.MIN_LEVEL=options.MIN_LEVEL||1;
 66 		/**
 67 		 * Maximum valid level
 68 		 * @public
 69 		 * @name MAX_LEVEL
 70 		 * @memberOf nanomaps.Projections.WebMercator#
 71 		 */
 72 		this.MAX_LEVEL=options.MAX_LEVEL||18;
 73 		
 74 		/**
 75 		 * Convert from global to projected units (x) 
 76 		 * @public
 77 		 * @name fwdX
 78 		 * @methodOf nanomaps.Projections.WebMercator#
 79 		 * @param {Number} x
 80 		 * @return {Number}
 81 		 */
 82 		this.fwdX=function(x) {
 83 			return x*DEG_TO_RAD * EARTH_RADIUS; 
 84 		};
 85 
 86 
 87 		/**
 88 		 * Convert from global to projected units (y) 
 89 		 * @public
 90 		 * @name fwdY
 91 		 * @methodOf nanomaps.Projections.WebMercator#
 92 		 * @param {Number} y
 93 		 * @return {Number}
 94 		 */
 95 		this.fwdY=function(y) {
 96 			return Math.log(Math.tan(FOURTHPI + 0.5 * DEG_TO_RAD * y)) * EARTH_RADIUS; 
 97 		};
 98 		
 99 		/**
100 		 * Convert from projected to global units (x) 
101 		 * @public
102 		 * @name invX
103 		 * @methodOf nanomaps.Projections.WebMercator#
104 		 * @param {Number} x
105 		 * @return {Number}
106 		 */
107 		this.invX=function(x) {
108 			return RAD_TO_DEG * x / EARTH_RADIUS;
109 		};
110 
111 		/**
112 		 * Convert from projected to global units (y) 
113 		 * @public
114 		 * @name invY
115 		 * @methodOf nanomaps.Projections.WebMercator#
116 		 * @param {Number} y
117 		 * @return {Number}
118 		 */
119 		this.invY=function(y) {
120 			return RAD_TO_DEG * (HALFPI - 2. * Math.atan(Math.exp(-y/EARTH_RADIUS)));
121 		};
122 		
123 		/**
124 		 * Return the resolution (m/px) for the given zoom index given a standard
125 		 * power of two zoom breakdown.
126 		 * @public
127 		 * @name fromLevel
128 		 * @methodOf nanomaps.Projections.WebMercator#
129 		 * @param {Number} level
130 		 * @return {Number} resolution
131 		 */
132 		this.fromLevel=function(level) {
133 			return HIGHEST_RES/Math.pow(2, level-1);
134 		};
135 		
136 		/**
137 		 * Return the level for the given resolution on the standard power of
138 		 * two scale.
139 		 * @public
140 		 * @name toLevel
141 		 * @methodOf nanomaps.Projections.WebMercator#
142 		 * @param {Number} resolution
143 		 * @return {Number} level
144 		 */
145 		this.toLevel=function(resolution) {
146 			return Math.log(HIGHEST_RES/resolution) / Math.log(2) + 1;
147 		};
148 
149 		/**
150 		 * Bounds object defining map extent in global
151 		 * coordinates.
152 		 * @public
153 		 * @name GLOBAL_EXTENT
154 		 * @memberOf nanomaps.Projections.WebMercator#
155 		 */
156 		var GLB_EXTENT=this.GLB_EXTENT=new Bounds(
157 			-180.0, -85.05112878, 180.0, 85.05112878
158 			);
159 			
160 		/**
161 		 * Bounds object defining map extent in projected
162 		 * coordinates
163 		 * @public
164 		 * @name GLOBAL_EXTENT
165 		 * @memberOf nanomaps.Projections.WebMercator#
166 		 */
167 		this.PRJ_EXTENT=new Bounds(
168 			this.fwdX(GLB_EXTENT.minx),
169 			this.fwdY(GLB_EXTENT.miny),
170 			this.fwdX(GLB_EXTENT.maxx),
171 			this.fwdY(GLB_EXTENT.maxy)
172 			);
173 		
174 		// release memory from closure
175 		options=null;
176 	}
177 };
178 
179 exports.Projections=Projections;
