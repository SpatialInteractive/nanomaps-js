  5 /**
  6  * Class that can handle motion events on a map
  7  * @constructor
  8  * @name nanomaps.MotionController
  9  * @private
 10  */
 11 var STATE_DOWN=0,
 12 	STATE_DRAG=1,
 13 	STATE_CLICK_PEND=2,
 14 	CLICK_DOUBLE_MS=280,
 15 	TOUCH_THRESHOLD=10,
 16 	TOUCH_LONGTAP_MS=1000,
 17 	TOUCH_DOUBLE_MS=280;
 18 
 19 /**
 20  * MotionEvent instances are passed to handler functions
 21  * and summarize touch or click interactions that are subject
 22  * to default handling.
 23  *
 24  * @public
 25  * @constructor
 26  * @name nanomaps.MotionEvent
 27  */
 28 function MotionEvent(type) {
 29 	/**
 30 	 * The type of motion event.  Different properties
 31 	 * will be available based on the type.
 32 	 * <ul>
 33 	 * <li>'click': A gesture to be interpreted as a click has
 34 	 * occurred.  The count field indicates how many consecutive clicks
 35 	 * took place.
 36 	 * <li>'drag': A single step in a drag gesture.  deltaX and deltaY
 37 	 * describe magnitude of this change
 38 	 * <li>'pinch': Single step of a pinch gesture.  This is a combined move
 39 	 * and zoom and has fields deltaX, deltaY and deltaZoom
 40 	 * <li>'longtap': A press and hold gesture has been detected.  If it is
 41 	 * handled, it will stop default processing which will either allow
 42 	 * panning/pinching or interpetation as a normal click/multi-click.
 43 	 * </ul>
 44 	 * @name type
 45 	 * @memberOf nanomaps.MotionEvent#
 46 	 */
 47 	this.type=type;
 48 	
 49 	/**
 50 	 * If the event has been handled by prior logic this flag
 51 	 * should be set to true, in which case, default actions will not
 52 	 * be performed.
 53 	 * @name handled
 54 	 * @memberOf nanomaps.MotionEvent#
 55 	 */
 56 	this.handled=false;
 57 }

 59 function MotionController(map) {
 60 	function dispatch(motionEvent) {
 61 		//console.log('dispatch motion event: ' + motionEvent.type + ': ' + motionEvent.count);
 62 		map.dispatchMotionEvent(motionEvent);
 63 	}

 65 	// ---- Click Handling
 66 	/**
 67 	 * If a click is in progress, this will be an object
 68 	 * {
 69 	 * }
 70 	 */
 71 	var clickState,
 72 		clickAttached;
 73 	
 74 	function clickAttach(attach) {
 75 		var document=map.elements.document;
 76 		if (attach && !clickAttached) {
 77 			addEventListener(document, 'mouseup', clickHandleEvent);
 78 			addEventListener(document, 'mousemove', clickHandleEvent);
 79 			clickAttached=true;
 80 		} else if (!attach && clickAttached) {
 81 			removeEventListener(document, 'mouseup', clickHandleEvent);
 82 			removeEventListener(document, 'mousemove', clickHandleEvent);
 83 			clickAttached=false;
 84 		}
 85 	}
 86 	
 87 	function clickStart(event, coords) {
 88 		clickCancel();
 89 		clickState={
 90 			b: event.button,
 91 			s: STATE_DOWN,	// State
 92 			cnt: 0,			// Click count
 93 			cs: coords,		// Click start coords
 94 			cl: coords,		// Click last coords
 95 			t: 0,			// Click time
 96 			ti: null		// Timer id
 97 		};
 98 	}
 99 	
100 	function clickStartTimer() {
101 		setTimeout(function() {
102 			if (clickState && clickState.s===STATE_CLICK_PEND) {
103 				clickState.ti=null;
104 				clickDispatch();
105 				clickCancel();
106 			}
107 		}, CLICK_DOUBLE_MS);
108 	}
109 	
110 	function clickStopTimer() {
111 		if (clickState && clickState.ti) {
112 			clearTimeout(clickState.ti);
113 		}
114 	}
115 	
116 	function clickDispatch() {
117 		//console.log('clickDispatch: ' + clickState.cnt);
118 		var me=new MotionEvent('click');
119 		
120 		/**
121 		 * Button number that this click represents as defined
122 		 * by W3C mouse events.
123 		 * <ul>
124 		 * <li>0 = Left / Primary
125 		 * <li>1 = Middle
126 		 * <li>2 = Right
127 		 * </ul>
128 		 * @public
129 		 * @name button
130 		 * @memberOf nanomaps.MotionEvent#
131 		 */
132 		me.button=clickState.b;
133 		
134 		/**
135 		 * For 'click' events, this is the number of consecutive
136 		 * clicks performed.  1 for single click, 2 for double click,
137 		 * 3 for lots of caffeine, etc.
138 		 * @public
139 		 * @name count
140 		 * @memberOf nanomaps.MotionEvent#
141 		 */
142 		me.count=clickState.cnt;
143 		
144 		/**
145 		 * Viewport x coordinate of the gesture.
146 		 * @public
147 		 * @name x
148 		 * @memberOf nanomaps.MotionEvent#
149 		 */
150 		me.x=clickState.cs.x;
151 
152 		/**
153 		 * Viewport y coordinate of the gesture.
154 		 * @public
155 		 * @name y
156 		 * @memberOf nanomaps.MotionEvent#
157 		 */
158 		me.y=clickState.cs.y;
159 		
160 		dispatch(me);
161 	}
162 	
163 	function clickCancel() {
164 		clickStopTimer();
165 		clickState=null;
166 	}
167 	
168 	/**
169 	 * This is the target of all click events
170 	 * (mousedown, mouseup, mousemove, mousewheel)
171 	 */
172 	function clickHandleEvent(event) {
173 		// Filter out any events that
174 		// are not the first button.
175 		// We'll enable other clicks later but they are tricky-tricky
176 		// to get right.
177 		if (!isLeftClick(event)) return;
178 		//console.log('handleClickEvent: ' + event.type + ' button=' + event.button);
179 		
180 		// It's ours now
181 		stopEvent(event);
182 		
183 		var coords=map.eventToContainer(event),
184 			me;
185 		
186 		switch (event.type) {
187 		case 'mousemove':
188 			if (clickState.s===STATE_DOWN || clickState.s===STATE_DRAG) {
189 				clickState.s=STATE_DRAG;
190 				me=new MotionEvent('drag');
191 				me.button=clickState.b;
192 				me.x=coords.x;
193 				me.y=coords.y;
194 				
195 				if (!clickState.cl) clickState.cl=clickState.cs;
196 				/**
197 				 * For drag, change in x from the last anchor position.
198 				 * @public
199 				 * @name deltaX
200 				 * @memberOf nanomaps.MotionEvent#
201 				 */
202 				me.deltaX=clickState.cl.x - coords.x;
203 
204 				/**
205 				 * For drag, change in y from the last anchor position.
206 				 * @public
207 				 * @name deltaY
208 				 * @memberOf nanomaps.MotionEvent#
209 				 */
210 				me.deltaY=clickState.cl.y - coords.y;
211 				clickState.cl=coords;
212 				
213 				dispatch(me);
214 			}
215 			break;
216 
217 		case 'mousedown':
218 			if (clickState && clickState.s===STATE_CLICK_PEND) {
219 				if ((now()-clickState.t)>CLICK_DOUBLE_MS) {
220 					// Not a valid followon click
221 					// Dispatch current and reset
222 					clickDispatch();
223 					clickStart(event, coords);
224 				} else {
225 					// Just let it resume the click sequence
226 					clickState.s=STATE_DOWN;
227 					clickStopTimer();
228 				}
229 			} else {
230 				clickStart(event, coords);
231 			}
232 			clickAttach(true);
233 			break;
234 			
235 		case 'mouseup':
236 			clickAttach(false);
237 			clickState.cnt++;
238 			if (clickState.s===STATE_DOWN) {
239 				// Start the double click timer
240 				clickState.s=STATE_CLICK_PEND;
241 				clickState.t=now();
242 				clickStartTimer();
243 			} else {
244 				clickCancel();
245 			}
246 			break;
247 			
248 		}
249 	}

251 	/**
252 	 * Handle mouse wheel events
253 	 * This is still shaky.  I think we need an interval timer
254 	 * and accumulator to straighten it out.  I also need a real
255 	 * scroll wheel mouse.
256 	 */
257 	function wheelHandleEvent(event) {
258 		stopEvent(event);
259 		clickCancel();
260 		clickAttach(false);
261 		
262 		var delta, factor=10.0,
263 			coords=map.eventToContainer(event), 
264 			me=new MotionEvent('scroll');
265 		if (event.wheelDelta) delta=event.wheelDelta/120;	// IE
266 		else if (event.delta) delta=-event.delta/3;
267 		else delta=-event.detail/3;
268 		
269 		// Wheel deltas can get pretty crazy.  Clamp them
270 		delta/=factor;
271 		if (delta>2.0) delta=2.0;
272 		else if (delta<-2.0) delta=-2.0;
273 		
274 		me.x=coords.x;
275 		me.y=coords.y;
276 		
277 		/**
278 		 * For 'scroll' events, this is the approximated zoom
279 		 * level change based on the magnitude of the scroll.
280 		 * @public
281 		 * @memberOf nanomaps.MotionEvent#
282 		 * @name deltaZoom
283 		 */
284 		me.deltaZoom=delta/factor;
285 		
286 		dispatch(me);
287 	}

289 	// -- Touch handling
290 	var touchState;
291 	
292 	function touchStartClickTimer() {
293 		touchStopTimer();
294 		touchState.ti=setTimeout(function() {
295 			touchState.ti=null;
296 			touchDispatchTap();
297 			touchCancel();
298 		}, TOUCH_DOUBLE_MS);
299 	}
300 	
301 	function touchStartLongTapTimer() {
302 		touchStopTimer();
303 		touchState.ti=setTimeout(function() {
304 			if (!touchState || touchState.s!==STATE_DOWN || touchState.t.length!==1) return;
305 			// Dispatch a longtap event and consume this tap if handled
306 			var me=new MotionEvent('longtap');
307 			me.button=0;
308 			me.x=touchState.t[0].x;
309 			me.y=touchState.t[0].y;
310 			me.count=1;
311 			dispatch(me);
312 			
313 			if (me.handled) {
314 				touchCancel();
315 			}
316 		}, TOUCH_LONGTAP_MS);
317 	}
318 	
319 	function touchStopTimer() {
320 		if (touchState) {
321 			if (touchState.ti) clearTimeout(touchState.ti);
322 			touchState.ti=null;
323 		}
324 	}
325 	
326 	function touchCancel() {
327 		touchStopTimer();
328 		touchState=null;
329 	}
330 	
331 	function touchDispatchTap() {
332 		var me=new MotionEvent('click');
333 		me.button=0;
334 		me.count=touchState.cnt;
335 		me.x=touchState.tapX;
336 		me.y=touchState.tapY;
337 		
338 		//console.log('tap: count=' + me.count + ', xy=(' + me.x + ',' + me.y + ')');
339 		dispatch(me);
340 	}

343 	function touchTranslate(eventTouches) {
344 		var touches=[];
345 		if (eventTouches) { 
346 			for (i=0; i<eventTouches.length; i++) {
347 				touch=eventTouches[i];
348 				touches.push(map.eventToContainer(touch));
349 			}
350 		}
351 		return touches;
352 	}

354 	function touchMoveSingle(currentXY, prevXY) {
355 		//console.log('touchMoveSingle (' + prevXY.x + ',' + prevXY.y + ') -> (' + currentXY.x + ',' + currentXY.y + ')');
356 		var me=new MotionEvent('drag');
357 		me.button=0;
358 		
359 		me.x=currentXY.x;
360 		me.y=currentXY.y;
361 		me.deltaX=prevXY.x - currentXY.x;
362 		me.deltaY=prevXY.y - currentXY.y;
363 		
364 		dispatch(me);
365 	}
366 	
367 	function touchMoveMulti(currentXY1, currentXY2, prevXY1, prevXY2) {
368 		//console.log('touchMoveMulti [(' + prevXY1.x + ',' + prevXY1.y + '),(' + prevXY2.x + ',' + prevXY2.y + ')] -> [(' + currentXY1.x + ',' + currentXY1.y + '),(' + currentXY2.x + ',' + currentXY2.y + ')]');
369 		
370 		var me=new MotionEvent('pinch'),
371 			ccX=(currentXY1.x+currentXY2.x)/2,
372 			ccY=(currentXY1.y+currentXY2.y)/2,
373 			pcX=(prevXY1.x+prevXY2.x)/2,
374 			pcY=(prevXY1.y+prevXY2.y)/2,
375 			cdx=currentXY1.x-currentXY2.x,
376 			cdy=currentXY1.y-currentXY2.y,
377 			pdx=prevXY1.x-prevXY2.x,
378 			pdy=prevXY1.y-prevXY2.y,
379 			cmag=Math.sqrt(cdx*cdx+cdy*cdy),
380 			pmag=Math.sqrt(pdx*pdx+pdy*pdy);
381 		
382 		me.button=0;
383 		me.x=ccX;
384 		me.y=ccY;
385 		me.deltaX=pcX-ccX;
386 		me.deltaY=pcY-ccY;
387 		
388 		/**
389 		 * For pinch, this is the change in zoom level that should be
390 		 * applied around the centroid (x,y) in addition to any movement
391 		 * by (deltaX, deltaY)
392 		 * @public
393 		 * @name deltaZoom
394 		 * @memberOf nanomaps.MotionEvent#
395 		 */
396 		me.deltaZoom=Math.log(cmag/pmag) * 1.5;
397 		
398 		//console.log('pinch: deltaXY=(' + me.deltaX + ',' + me.deltaY + '), deltaZoom=' + me.deltaZoom);
399 		dispatch(me);
400 	}

402 	function touchHandleEvent(event) {
403 		var type=event.type,
404 			currentTouches;
405 			
406 		stopEvent(event);
407 		
408 		// Extract current touches and translate to local coordinates
409 		currentTouches=touchTranslate(event.touches);
410 		
411 		//console.log('ontouch: ' + event.type + ', touches=' + currentTouches.length);
412 		
413 		// Handle touch start
414 		if (type==='touchstart') {
415 			// Handle followon tap
416 			if (touchState && touchState.s===STATE_CLICK_PEND) {
417 				if ((now()-touchState.te)>TOUCH_DOUBLE_MS) {
418 					// Not a valid followon tap
419 					// Dispatch current and reset
420 					touchDispatch();
421 					touchCancel();
422 					
423 					// Fall through to normal touch start
424 				} else if (currentTouches.length===1) {
425 					// Just let it resume the tap-tap sequence
426 					touchState.s=STATE_DOWN;
427 					touchState.t=currentTouches;
428 					touchStopTimer();
429 					return;
430 				}
431 			}
432 			
433 			// Normal touch initiation when a new finger is pressed
434 			if (!touchState) {
435 				touchState={
436 					s: STATE_DOWN,
437 					t: currentTouches,	// Touches array
438 					mt: false,			// True if multi-touch
439 					te: 0,				// Last touch end time
440 					ts: now(),			// Current touch start time
441 					ti: null,			// Touch timer id
442 					cnt: 0				// Number of consecutive touches
443 				};
444 				touchStartLongTapTimer();
445 			} else {
446 				if (currentTouches.length>1) touchState.mt=true;
447 				touchState.t=currentTouches;
448 			}
449 			return;
450 		} 
451 		
452 		// Don't do anything else if not in a touch state
453 		if (!touchState) return;
454 		
455 		switch (event.type) {
456 		case 'touchmove':
457 			if (currentTouches.length===1) {
458 				if (touchState.s===STATE_DRAG ||
459 					Math.abs(currentTouches[0].x-touchState.t[0].x)>TOUCH_THRESHOLD ||
460 					Math.abs(currentTouches[0].y-touchState.t[0].y)>TOUCH_THRESHOLD)
461 				{
462 					// Cancel any longtap in progress
463 					touchStopTimer();
464 					
465 					// Single touch over touch threshold
466 					if (touchState.s===STATE_CLICK_PEND) {
467 						// Dispatch pending and reset
468 						touchDispatch();
469 						touchState.cnt=0;
470 					}
471 					
472 					touchState.s=STATE_DRAG;
473 					touchMoveSingle(currentTouches[0], touchState.t[0]);
474 					touchState.t=currentTouches;
475 				}
476 			} else if (currentTouches.length>1) {
477 				touchState.s=STATE_DRAG;
478 				touchMoveMulti(
479 					currentTouches[0],
480 					currentTouches[1],
481 					touchState.t[0],
482 					touchState.t[1]
483 				);
484 				touchState.t=currentTouches;
485 			}
486 			
487 			break;
488 		case 'touchend':
489 			// Note that ios calls touchend with 0 touches
490 			// when done, but android calls it with its 1 touch
491 			// still present (at least on single touch devices).
492 			// This is a pita since we don't have a clear signal
493 			// of completion.
494 			// This may not work on android multi-touch.
495 			if (!touchState.mt || currentTouches.length===0) {
496 				// All done here
497 				if (touchState.s===STATE_DOWN) {
498 					touchState.cnt++;
499 					touchState.s=STATE_CLICK_PEND;
500 					touchState.tapX=touchState.t[0].x;
501 					touchState.tapY=touchState.t[0].y;
502 					touchState.te=now();
503 					touchStartClickTimer();
504 				} else {
505 					touchCancel();
506 				}
507 			} else {
508 				touchState.t=currentTouches;
509 			}
510 			break;
511 		case 'touchcancel':
512 			touchCancel();
513 			break;
514 		}
515 	}

517 	/**
518 	 * Listen to all motion related events on
519 	 * the map.
520 	 *
521 	 * @methodOf Nanomaps.MotionController#
522 	 */
523 	this.listen=function(enableClick, enableTouch, enableWheel) {
524 		var elements=map.elements,
525 			target=elements.event,
526 			parent=elements.parent,
527 			touchEvents=[
528 				'touchstart',
529 				'touchend',
530 				'touchmove',
531 				'touchcancel'
532 			],
533 			i;
534 		
535 		if (enableClick) {
536 			// Just listen for mousedown to start with
537 			// We will listen for mousemove and mouseup
538 			// on document when we enter a down state
539 			addEventListener(target, 'mousedown', clickHandleEvent);
540 			
541 			// Also listen on the parent so that we get mousedowns
542 			// that bubble up the hierarchy from inactive elements
543 			// ** Actually not doing this as it completely breaks containment
544 			//addEventListener(parent, 'mousedown', clickHandleEvent);
545 		}
546 		
547 		if (enableWheel) {
548 			addEventListener(target, 'DOMMouseScroll', wheelHandleEvent);
549 			addEventListener(target, 'mousewheel', wheelHandleEvent);
550 		}
551 		
552 		if (enableTouch && target.addEventListener) {
553 			for (i=0; i<touchEvents.length; i++) {
554 				target.addEventListener(touchEvents[i], touchHandleEvent, true);
555 			}
556 		}
557 	};

580 /**
581  * Dispatch a MotionEvent for processing.  This will emit
582  * the 'motion' event on the map and then call handleMotionEvent
583  * for default processing if the handled flag is false.
584  * @public
585  * @name dispatchMotionEvent
586  * @methodOf nanomaps.MapSurface.prototype
587  * @param motionEvent {nanomaps.MotionEvent}
588  */
589 MapSurfaceMethods.dispatchMotionEvent=function(motionEvent) {
590 	var eventName='motion.' + motionEvent.type;
591 	//console.log('emit(' + eventName + ')');
592 	this.emit(eventName, motionEvent);
593 	if (!motionEvent.handled) {
594 		this.handleMotionEvent(motionEvent);
595 	}
596 };
597 
598 /**
599  * Perform default processing of a MotionEvent if its handled
600  * property is false after dispatch.  You can intercept motion
601  * events by registering an event with name 'motion.{type}'.
602  * <pre>
603  * 	map.on('motion.click', function(motionEvent) {
604  * 		alert('Clicked at ' + motionEvent.x + ',' + motionEvent.y);
605  * 		motionEvent.handled=true;
606  * 	});
607  * </pre>
608  * @public
609  * @name handleMotionEvent
610  * @methodOf nanomaps.MapSurface.prototype
611  * @param motionEvent {nanomaps.MotionEvent}
612  */
613 MapSurfaceMethods.handleMotionEvent=function(motionEvent) {
614 	if (motionEvent.handled) return;
615 	
616 	var type=motionEvent.type, deltaZoom;
617 	if (type==='drag' || type==='pinch') {
618 		this.begin();
619 		this.moveBy(motionEvent.deltaX, -motionEvent.deltaY);
620 		deltaZoom=Number(motionEvent.deltaZoom);
621 		if (!isNaN(deltaZoom)) {
622 			this.setZoom(this.getZoom() + deltaZoom, motionEvent.x, motionEvent.y);
623 		}
624 		this.commit();
625 		motionEvent.handled=true;
626 	} else if (type==='scroll') {
627 		this.setZoom(this.getZoom()+motionEvent.deltaZoom, 
628 			motionEvent.x, motionEvent.y);
629 		motionEvent.handled=true;
630 	} else if (type==='click' && motionEvent.count===2) {
631 		// Double-click to zoom
632 		this.begin();
633 		this.zoomIn(motionEvent.x, motionEvent.y);
634 		this.commit(true);
635 		motionEvent.handled=true;
636 	}
637 };
