  6 // Helpers to get x/y alias names
  7 var X_NAMES=['x', 'lng', 'longitude', 'lon'],
  8 	Y_NAMES=['y', 'lat', 'latitude'];
  9 function firstNumeric(object, props) {
 10 	var i,v,t;
 11 	for (i=0; i<props.length; i++) {
 12 		v=object[props[i]];
 13 		if (v!==undefined && v!==null) {
 14 			t=typeof v;
 15 			if (t==='number') return v;
 16 			else if (t==='function') return Number(v.call(object));
 17 		}
 18 	}
 19 	return NaN;
 20 }

 22 /**
 23  * Abstracts out a single point in a global coordinate system.
 24  * This may be neurotic, but my brain is wired to think of (latitude,longitude)
 25  * and when its in that mode, I can't get it in my head that x=longitude and
 26  * y=latitude (in fact, just after writing this sentence I had to read it 3 times
 27  * because that didn't quite look right).  
 28  * As a further neurosis, I spent my productive map-engaged years abbreviating them (lat,lng).
 29  * <p>
 30  * So the presence of this class is a tip of the hat to the fact that no one
 31  * can cure the brain and quirky habits are better indulged in old age than
 32  * challenged.  Everyone else can just go on using (x,y) and remembering
 33  * the difference.  I name things explicitly here so as not to get confused.  Surely
 34  * a lifetime of abuse is worth a few bytes in a library.
 35  * <p>
 36  * The entire rest of the library deals in (x,y).  I've tried to keep any concept
 37  * of latitude and longitude here in the form of helper methods.  The "public" api
 38  * should pass around instances of this class.  The bits under the covers, though
 39  * will often just passes x's and y's nakedly (ie. see MapState).
 40  * <p>
 41  * This class is also structured to allow some future refactoring to support double
 42  * or integral E6 coordinates.  This could save some space with lots instances (maybe,
 43  * depending on alignment) but my intuition tells me it is probably not a speed savings
 44  * since we've got to go back to double anyway for projection math.  In my experience
 45  * this is usually a premature optimization, but I leave the option open by way of a subclass.
 46  * <p>
 47  * Ok, that's enough of a treatise on a simple Coordinate class.
 48  *
 49  * @constructor
 50  * @name nanomaps.Coordinate
 51  * @param x horizontal coordinate
 52  * @param y vertical coordinate
 53  * @author stella
 54  */
 55 function Coordinate(x,y) {
 56 	this._x=Number(x);
 57 	this._y=Number(y);
 58 }
 59 /**
 60  * Static function for returning a coordinate given
 61  * latitude,longitude.  This is here just to make sure
 62  * I get it right instead of relying on my memory that
 63  * lat=y and lng=x.
 64  * @public
 65  * @name latLng
 66  * @methodOf nanomaps.Coordinate
 67  * @param lat {Number}
 68  * @param lng {Number}
 69  * @return {Coordinate}
 70  */
 71 Coordinate.latLng=function(lat, lng) {
 72 	return new Coordinate(lng, lat);
 73 };

 88 /**
 89  * Generic conversion function to produce a Coordinate
 90  * from a variety of objects.  If object is a Coordinate,
 91  * then object is returned as-is.  Otherwise, the x-coordinate
 92  * is determined by the first existent property:
 93  * <ol>
 94  * <li>x
 95  * <li>lng
 96  * <li>longitude
 97  * <li>lon
 98  * </ol>
 99  * The y coordinate is the first property:
100  * <ol>
101  * <li>y
102  * <li>lat
103  * <li>latitude
104  * </ol>
105  * 
106  * Alternatively, if object is an array, then it is
107  * taken to be [x, y]
108  * @public
109  * @name from
110  * @methodOf nanomaps.Coordinate
111  * @param object {Object}
112  * @return {Coordinate}
113  */
114 Coordinate.from=function(object) {
115 	if (object instanceof Coordinate) {
116 		return object;
117 	}
118 	if (object instanceof Array) {
119 		return new Coordinate(object[0], object[1]);
120 	}
121 	var x=firstNumeric(object, X_NAMES),
122 		y=firstNumeric(object, Y_NAMES);
123 	return new Coordinate(x,y);
124 };

197 /**
198  * Convert tile x,y,level coordinates to a Microsoft
199  * quad-key
200  */
201 function tileXYToQuadkey(tileX, tileY, level) {
202 	var i, mask, value, ret='';
203 	for (i=level; i>0; i--) {
204 		value=48;
205 		mask=1<<(i-1);
206 		if ((tileX&mask)!==0) value++;
207 		if ((tileY&mask)!==0) value+=2;
208 		ret+=String.fromCharCode(value);
209 	}
210 	return ret;
211 }
