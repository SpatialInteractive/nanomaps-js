1 /**
  2  * nanomaps.imgmarker.js
  3  * Factories for constructing HTML fragments suitable to be used
  4  * as markers from stock imagery
  5  */
  6 function makeAbsolute(baseUri, href) {
  7 	if (href[0]==='/' || href.search(/^http(s)?\:/)>=0) return href;
  8 	if (baseUri.match(/\/$/)) return baseUri+href;
  9 	else return baseUri+'/'+href;
 10 }
 11 
 12 function hrefToClassSuffix(href) {
 13 	var i=href.lastIndexOf('/'),
 14 		baseName=(i>=0 ? href.substring(i+1) : href),
 15 		m=baseName.match(/^([^_\-\.]+)/);
 16 	return m ? m[0] : href;
 17 }

 19 /**
 20  * Primordial image marker class.  Image markers are managed according
 21  * to the prototype pattern.  If you have one, you can customize it into
 22  * a new one.  Any marker instance can be used in a call to MapSurface.attach
 23  * due to it fulfilling the element factory interface.
 24  * @public
 25  */
 26 function ImgMarker(settings) {
 27 	this.settings=new SettingsConstructor(settings);
 28 	this.element=null;
 29 	this.children=null;
 30 }
 31 /**
 32  * Global settings inherited for each instance
 33  */
 34 ImgMarker.Settings={
 35 	/**
 36 	 * The img src.  If primarySuffix is not specified,
 37 	 * then the first alpha-numeric part of this string
 38 	 * is taken.
 39 	 */
 40 	src: 'orb_blue.png',
 41 	
 42 	/**
 43 	 * The uri to resolve relative sources against
 44 	 */
 45 	baseUri: '',
 46 	
 47 	/**
 48 	 * Location Coordinate
 49 	 */
 50 	location: null,
 51 	
 52 	/**
 53 	 * If set, then this is the primary css suffix.  If not set,
 54 	 * it is derived from the src.
 55 	 */
 56 	 classSuffix: '',
 57 	 
 58 	 /**
 59 	  * Extra css classes
 60 	  */
 61 	 extraClasses: ''
 62 };
 63 
 64 ImgMarker.prototype={
 65 	/**
 66 	 * By default ImgMarker instances are attached to the 'overlay' layer
 67 	 * @public
 68 	 * @name defaultLayer
 69 	 * @memberOf nanomaps.ImgMarker.prototype
 70 	 */
 71 	defaultLayer: 'overlay',
 72 
 73 	/**
 74 	 * Update the latitude/longitude settings.  If the marker is attached,
 75 	 * make the updates live on the map as well.
 76 	 * @methodOf nanomaps.ImgMarker.prototype
 77 	 * @name setLocation
 78 	 * @param coordinate {Coordinate}
 79 	 */
 80 	setLocation: function(coordinate) {
 81 		coordinate=Coordinate.from(coordinate);
 82 		var settings=this.settings,
 83 			element=this.element,
 84 			owner=this.owner;
 85 		settings.location=coordinate;
 86 		if (element) {
 87 			if (coordinate) {
 88 				element.geo={
 89 					latitude: coordinate.lat(),
 90 					longitude: coordinate.lng()
 91 				};
 92 			} else {
 93 				element.geo=null;
 94 			}
 95 			if (owner&&element.parentNode) {
 96 				owner.update(element);
 97 			}
 98 		}
 99 	},
100 	
101 	/**
102 	 * (Attachment Api) Get the element that should be attached
103 	 * to the map.
104 	 * <pre>
105 	 * <div class="nmim">
106 	 *   <img src="{src url}" class="nmim-fg {src class}"/>
107 	 * </div>
108 	 * </pre>
109 	 * A background will only be specified if setup in the properties.  In addition,
110 	 * the actual dom element returned will have its "geo" object set so that
111 	 * the default map delegate can position it appropriately.
112 	 * @public
113 	 * @memberOf nanomaps.ImgMarker.prototype
114 	 */
115 	getElement: function(mapSurface) {
116 		this.owner=mapSurface;
117 		if (this.element) return this.element;
118 		var d=mapSurface.elements.document,
119 			element=d.createElement('div'),
120 			fgElt=d.createElement('img');
121 		element.appendChild(fgElt);
122 		
123 		// Export
124 		this.element=element;
125 		this.children={
126 			fg: fgElt
127 		};
128 		
129 		// Apply config
130 		this._config();
131 		return element;
132 	},
133 	
134 	/**
135 	 * Apply the current configuration to the elements
136 	 * @private
137 	 * @name _config
138 	 * @memberOf nanomaps.ImgMarker.prototype
139 	 */
140 	_config: function() {
141 		var settings=this.settings,
142 			location=settings.location,
143 			element=this.element,
144 			fgElt=this.children.fg,
145 			fgSrc,
146 			classSuffix;
147 			
148 			
149 		// Configure fg img elt
150 		fgSrc=makeAbsolute(settings.baseUri||'', settings.src||'');
151 		classSuffix=settings.classSuffix || hrefToClassSuffix(fgSrc);
152 		
153 		element.className='nmim nmim-' + classSuffix + ' ' + settings.extraClasses;
154 		if (location) {
155 			element.geo={
156 				latitude: location.lat()||NaN,
157 				longitude: location.lng()||NaN
158 			};
159 		} else {
160 			element.geo=null;
161 		}
162 		
163 		fgElt.className='nmim-fg';
164 		fgElt.setAttribute('src', fgSrc||null);
165 	}
166 };
167 
168 /**
169  * Used internally to create a new settings object with the global
170  * settings as its prototype.
171  * @private
172  */
173 function SettingsConstructor(override) {
174 	if (override) {
175 		for (var k in override) {
176 			if (override.hasOwnProperty(k))
177 				this[k]=override[k];
178 		}
179 	}
180 }
181 SettingsConstructor.prototype=ImgMarker.Settings;
