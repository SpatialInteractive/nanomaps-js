1 /**
  2  * Extend MapSurface with standard support for desktop event handling:
  3  *  - mouse events
  4  *  - mouse cursors
  5  *  - translation of desktop events to gestures
  6  */
  7 (function(nanomaps) {
  8 	var MapSurfaceMethods=nanomaps.MapSurface.prototype;
  9 	
 10 	/**
 11 	 * Return true if the device may be touch enabled.  This does not actually
 12 	 * detect if touch is absolutely supported but filters out legacy browsers
 13 	 * that won't even run the code.
 14 	 */
 15 	function touchEligible() {
 16 		return !!document.addEventListener;
 17 	}
 18 	
 19 	function getTouchState(map) {
 20 		return map._touchState;
 21 	}
 22 	
 23 	function recordTouches(map, touchState, event) {
 24 		var i, touch, touches=event.touches, dest=[];
 25 		for (i=0; i<touches.length; i++) {
 26 			touch=touches[i];
 27 			dest.push(map.eventToContainer(touch));
 28 		}
 29 		
 30 		touchState.t=dest;
 31 	}
 32 	
 33 	function clearTouchState(touchState) {
 34 		touchState.s='';
 35 		touchState.l=null;
 36 		touchState.c=null;
 37 		touchState.z=null;
 38 		if (touchState.iid) {
 39 			clearInterval(touchState.iid);
 40 			touchState.iid=null;
 41 		}
 42 		if (touchState.tkey) {
 43 			clearTimeout(touchState.tkey);
 44 			touchState.tkey=null;
 45 		}
 46 	}
 47 	
 48 	function calculateCenter(touchState) {
 49 		var i, x=0, y=0;
 50 		for (i=0; i<touchState.t.length; i++) {
 51 			x+=touchState.t[i].x;
 52 			y+=touchState.t[i].y;
 53 		}
 54 		x/=i;
 55 		y/=i;
 56 		
 57 		return {x:x,y:y};
 58 	}
 59 	
 60 	function clampMove(delta) {
 61 		var absDelta=Math.abs(delta), sign=delta/absDelta, limit=100;
 62 		if (absDelta>250 || absDelta<limit) return delta;
 63 		return limit*sign;
 64 	}
 65 	
 66 	function startInterval(map, touchState) {
 67 		if (touchState.iid) clearInterval(touchState.iid);
 68 		touchState.iid=setInterval(function() {
 69 			var	northing=clampMove(touchState.dn), easting=clampMove(touchState.de);
 70 			if (northing || easting) {
 71 				map.moveBy(easting, northing);
 72 				touchState.dn-=northing;
 73 				touchState.de-=easting;
 74 			}
 75 		}, 65);
 76 	}

 91 	MapSurfaceMethods._handleTouch=function(event) {
 92 		event.preventDefault();
 93 		//console.log('[touch event]: ' + event.type);
 94 		
 95 		var self=this,
 96 			touchState=this._touchState,
 97 			existingTouches=touchState.t.length,
 98 			deltaX, deltaY,
 99 			startZoom, newZoom;
100 		
101 		
102 		switch (event.type) {
103 		case 'touchmove':
104 			recordTouches(this, touchState, event);
105 			if ((touchState.s==='move' || touchState.s==='start') && touchState.t.length===1) {
106 				touchState.s='move';
107 				touchState.de+=touchState.l.x - touchState.t[0].x;
108 				touchState.dn+=touchState.t[0].y - touchState.l.y; // invert northing
109 				touchState.l=touchState.t[0];
110 				
111 				//this.moveBy(touchState.de, touchState.dn);
112 			}
113 			
114 			break;
115 		
116 		case 'gesturechange':
117 			// If not in gesture state, find and stash the center point
118 			if (touchState.s!=='gesture') {
119 				touchState.c=calculateCenter(touchState);
120 				touchState.z=map.getLevel();
121 			}
122 	
123 			// Upgrade to a gesture state (prevents moving)
124 			touchState.s='gesture';
125 			
126 			startZoom=touchState.z;
127 			newZoom=startZoom+Math.log(event.scale)*1.5;
128 			//console.log('Scale OrigLevel=' + touchState.z + ', newLevel=' + newZoom + ' scale=' + event.scale);
129 			//if (touchState.tkey) clearTimeout(touchState.tkey);
130 			//touchState.tkey=setTimeout(function() {
131 				self.setLevel(newZoom, touchState.c);
132 				//touchState.tkey=null;
133 			//}, 0);
134 		
135 			break;
136 			
137 		case 'touchstart':
138 			recordTouches(this, touchState, event);
139 			if (existingTouches==0 && touchState.t.length) {
140 				// Transition from no touches to a first touch
141 				touchState.s='start';
142 				
143 				touchState.dn=0;	// Delta-northing
144 				touchState.de=0;	// Delta-easting
145 				
146 				startInterval(this, touchState);
147 				
148 				// Reset last (for moving)
149 				touchState.l=touchState.t[0];
150 			}			
151 			break;
152 		
153 		case 'touchend':
154 			recordTouches(this, touchState, event);
155 			if (!touchState.t.length) clearTouchState(touchState);
156 			if (touchState.de||touchState.dn) this.moveBy(touchState.de, touchState.dn);
157 			break;
158 			
159 		case 'gestureend':
160 			clearTouchState(touchState);
161 			break;
162 			
163 		case 'touchmove':
164 			recordTouches(this, touchState, event);
165 			if ((touchState.s==='move' || touchState.s==='start') && touchState.t.length===1) {
166 				touchState.s='move';
167 				deltaX=touchState.l.x - touchState.t[0].x;
168 				deltaY=touchState.l.y - touchState.t[0].y;
169 				touchState.l=touchState.t[0];
170 				
171 				this.moveBy(deltaX, -deltaY);	// Invert northing
172 			}
173 			
174 			break;
175 			
176 		case 'touchcancel':
177 			clearTouchState(touchState);
178 			break;
179 		
180 		}
181 	};
182 	
183 	function touchEventDispatcher(map) {
184 		return function(event) {
185 			return map._handleTouch(event);
186 		};
187 	}
188 	
189 	// Attach to initialization
190 	if (touchEligible()) {
191 		MapSurfaceMethods.advise('initialize', 'after', function(options) {
192 			this._touchState={t:[]};
193 			var events=['touchstart', 'touchend', 'touchmove', 'touchcancel',
194 						'gesturechange', 'gestureend' ],
195 				listener=touchEventDispatcher(this);
196 				
197 			// NOTE: We must attach touch events to the glass element.  Webkit
198 			// seems to randomly cancel touch events if any DOM structure
199 			// in the containment hierarchy gets modified "too much".  The only
200 			// way I could find around this was to use such a transparent overlay.
201 			// We're going to need to pump events back to the source later.
202 			this.elements.glass.style.display='block';
203 			for (i=0; i<events.length; i++) {
204 				this.elements.glass.addEventListener(events[i], listener, true);
205 			}
206 		});
207 	}
208 })(nanomaps);
