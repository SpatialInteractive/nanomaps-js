 13 function makeVisibleOnLoad() {
 14 	this.style.visibility='';

 17 function TileCache() {
 18 	var contents={},
 19 		marked=[];
 20 	
 21 	function disposeTileDesc(tileDesc) {
 22 		if (tileDesc.img&&tileDesc.img.parentNode) 
 23 			// Detach
 24 			tileDesc.img.parentNode.removeChild(tileDesc.img);
 25 	}
 26 		
 27 	this.mark=function() {
 28 		marked.length=0;
 29 	};
 30 	
 31 	this.free=function() {
 32 		var newContents={},
 33 			key, i, tileDesc;
 34 			
 35 		// Add all marked to the new dictionary
 36 		for (i=0; i<marked.length; i++) {
 37 			tileDesc=marked[i];
 38 			key=tileDesc.key;
 39 			newContents[key]=tileDesc;
 40 			delete contents[key];
 41 		}
 42 		
 43 		// Any existing in contents are to be discarded
 44 		for (i in contents) {
 45 			tileDesc=contents[i];
 46 			disposeTileDesc(tileDesc);
 47 		}
 48 		
 49 		// Swap
 50 		contents=newContents;
 51 		marked.length=0;
 52 	};
 53 	
 54 	this.use=function(tileDesc) {
 55 		var key=tileDesc.key,
 56 			existing=contents[key];
 57 		if (existing) {
 58 			disposeTileDesc(tileDesc);
 59 			tileDesc=existing;
 60 		}
 61 		contents[key]=tileDesc;
 62 		if (marked) marked.push(tileDesc);
 63 		return tileDesc;
 64 	};
 65 	
 66 	this.take=function(tileDesc) {
 67 		var key=tileDesc.key,
 68 			existing=contents[key];
 69 		if (existing) {
 70 			delete contents[key];
 71 			return existing;
 72 		}
 73 		return tileDesc;
 74 	};
 75 	
 76 	this.each=function(callback) {
 77 		var i, tileDesc;
 78 		for (i in contents) {
 79 			tileDesc=contents[i];
 80 			if (tileDesc && typeof tileDesc==='object')
 81 				callback(tileDesc);
 82 		}
 83 	};
 84 	
 85 	this.clear=function() {
 86 		contents={};
 87 		marked.length=0;
 88 	};
 89 	
 90 	this.moveFrom=function(otherCache) {
 91 		otherCache.each(function(tileDesc) {
 92 			var key=tileDesc.key;
 93 			if (contents[key]) {
 94 				// Already exists.  Delete
 95 				disposeTileDesc(tileDesc);
 96 			} else
 97 				contents[tileDesc.key]=tileDesc;
 98 		});
 99 		otherCache.clear();
100 	};
101 }

103 /**
104  * Construct a standard TileLayer that can be attached to a MapSurface.  If a
105  * TileSelector is not passed in the options, then a default is constructed and
106  * options are passed to it.  See the options accepted by the TileSelector
107  * constructor for additional parameters.
108  *
109  * @example 
110  * var map=new nanomaps.MapSurface(someElement);
111  * map.attach(new nanomaps.TileLayer({ 
112  *    tileSrc: "http://otile${modulo:1,2,3}.mqcdn.com/tiles/1.0.0/osm/${level}/${tileX}/${tileY}.png" })); 
113  *
114  * @class
115  * @name nanomaps.TileLayer
116  * @param {TileSelector} [options.selector] describes the source and geometry of the
117  * tiles
118  * @param {integer} [options.buffer=64] The number of pixels to actively buffer on
119  * all sides of the map
120  */
121 function TileLayer(options) {
122 	this.options=options=options||{};
123 	this.sel=options.selector||new TileSelector(options);
124 	this.fgCache=new TileCache();
125 	this.bgCache=new TileCache();
126 }
127 TileLayer.prototype={
128 	/**
129 	 * Returns a new element ready to be added to the MapSurface.  This method
130 	 * fulfills the contract for MapSurface.attach.
131 	 * @public
132 	 * @methodOf nanomaps.TileLayer.prototype
133 	 * @name createElement
134 	 * @param {MapSurface} map
135 	 * @return {HTMLElement}
136 	 */
137 	createElement: function(map) {
138 		var element=document.createElement('div');
139 		element.style.position='absolute';
140 		element.style.left='0px';
141 		element.style.top='0px';
142 		element.mapDelegate=this;
143 		return element;
144 	},
145 	
146 	/**
147 	 * @private
148 	 */
149 	placeTile: function(map, element, tileDesc, moveToFront) {
150 		var img=tileDesc.img,
151 			transform=map.transform,
152 			zpx=transform.zpx,
153 			scaleFactor=tileDesc.res / transform.res;
154 			
155 		if (img&&img._error) {
156 			// Zero it out
157 			if (img.parentElement) img.parentElement.removeChild(img);
158 			tileDesc.img=null;
159 			img=null;
160 		}
161 		
162 		// If no valid img, instantiate it
163 		if (!img) {
164 			img=map.createElement('img');
165 			img.style.visibility='hidden';
166 			img.onload=makeVisibleOnLoad;
167 			img.style.position='absolute';
168 			img.src=this.sel.resolveSrc(tileDesc);
169 			tileDesc.img=img;
170 		}
171 		
172 		// Set position and size
173 		img.width=Math.ceil(tileDesc.size*scaleFactor);
174 		img.height=Math.ceil(tileDesc.size*scaleFactor);
175 		img.style.left=Math.round(tileDesc.x*scaleFactor - zpx[0]) + 'px';
176 		img.style.top=Math.round(zpx[1] - tileDesc.y*scaleFactor) + 'px';	// y-axis inversion
177 		if (moveToFront || img.parentNode!==element) element.appendChild(img);
178 	},
179 	
180 	/**
181 	 * @private
182 	 */
183 	onreset: function(map, element) {
184 		var self=this,
185 			transform=map.transform,
186 			buffer=self.options.buffer||64,
187 			ulXY=map.toGlobalPixels(-buffer,-buffer),
188 			width=map.width+buffer,
189 			height=map.height+buffer,
190 			curResolution=self.curResolution,
191 			displayResolution=transform.res,
192 			tileList=self.sel.select(transform.prj, displayResolution, ulXY.x, ulXY.y, width, height, true),
193 			fgCache=self.fgCache,
194 			bgCache=self.bgCache,
195 			refreshBackground=false,
196 			i,
197 			tileDesc;
198 			
199 		if (curResolution&&curResolution!==displayResolution) {
200 			// We have a scale change.  Snapshot the current tile
201 			// cache as the background cache and start on a new one
202 			bgCache.mark();
203 			bgCache.free();
204 			bgCache.moveFrom(fgCache);
205 			refreshBackground=true;
206 		}
207 
208 		fgCache.mark();
209 		for (i=0; i<tileList.length; i++) {
210 			tileDesc=fgCache.use(bgCache.take(tileList[i]));
211 			self.placeTile(map, element, tileDesc, true);
212 		}
213 		fgCache.free();
214 
215 		// Record so that the next time through we can tell whether
216 		// this is a simple change
217 		this.curResolution=displayResolution;
218 		
219 		if (refreshBackground) {
220 			bgCache.each(function(tileDesc) {
221 				self.placeTile(map, element, tileDesc);
222 			});
223 		}
224 		
225 	},
226 	
227 	/**
228 	 * @private
229 	 */
230 	onposition: function(map, element) {
231 		this.onreset(map, element);
232 	}
233 };

235 /**
236  * Class representing a tile selector for the layout of OSM, MS, Google, et al.
237  * <p>
238  * Typically, all that will need to be specified is options.tileSrc.  This string
239  * will be interpolated relative to the tile information with the following
240  * substitutions made:
241  * <ul>
242  * <li>${level} - Native level of the tile
243  * <li>${tileX} - X coordinate of the tile
244  * <li>${tileY} - Y coordinate of the tile
245  * <li>${modulo:a,b} - Select one of the values in the list (in this case 'a' and 'b')
246  *     based on a stable modulus based on the tile coordinate system.  Used to
247  *     pick a domain name for a given tile.
248  * </ul>
249  * <p>
250  * Remember to always provide proper attribution and get permission for using tiles.
251  * <p>
252  * The following tileSrc can be used to display tiles from different providers:
253  * <ul>
254  * <li>OSM (MapQuest): 
255  * 		"http://otile${modulo:1,2,3}.mqcdn.com/tiles/1.0.0/osm/${level}/${tileX}/${tileY}.png"
256  * <li>OSM (Mapnik from openstreetmap.org): "http://${modulo:a,b,c}.tile.openstreetmap.org/${level}/${tileX}/${tileY}.png"
257  * </ul>
258  *
259  * @class
260  * @name nanomaps.TileSelector
261  * @param {integer} [options.tileSize=256]
262  * @param {string} [options.tileSrc='']
263  */
264 function TileSelector(options) {
265 	options=options||{};
266 	// set defaults
267 	this.tileSize=options.tileSize||256;
268 	this.srcSpec=options.tileSrc||'';
269 }
270 TileSelector.prototype={
271 	/**
272 	 * Gets the origin in physical units [x, y] for the tile coordinate space
273 	 * @private
274 	 */
275 	origin: function(projection) {
276 		var ret=this._origin;
277 		if (!ret) {
278 			ret=projection.forward(-180, 85.05112878);
279 			this._origin=ret;
280 		}
281 		return ret;
282 	},
283 	
284 	/**
285 	 * Get an img src from the tileDesc
286 	 * @public
287 	 * @name resolveSrc
288 	 * @methodOf nanomaps.TileSelector.prototype
289 	 * @param {TileDesc} tileDesc tile description returned from select
290 	 */
291 	resolveSrc: function(tileDesc) {
292 		return this.srcSpec.replace(/\$\{([A-Za-z]+)(\:([^\}]*))?\}/g, function(s, name, ignore, args) {
293 			if (name==='modulo') {
294 				// get the args and modulo it by the tileDesc.tileX
295 				args=args.split(/\,/);
296 				return args[tileDesc.tileX%args.length];
297 			} else {
298 				return tileDesc[name];
299 			}
300 		});
301 	},
302 	
303 	/**
304 	 * Select all tiles that intersect the given bounding box at the
305 	 * given resolution.  Return an array of TileDesc elements, where
306 	 * each TileDesc has the following attributes:
307 	 * <ul>
308 	 *   <li>key: an opaque string that uniquely identifies this tile within
309 	 *     this instance where multiple requests for the same physical tile
310 	 *     have the same key
311 	 *   <li>res: the native resolution of the tile
312 	 *	 <li>level: the native zoom level
313 	 *   <li>x, y: origin at native resolution,
314 	 *	 <li>size: tile size (width/height) at native resolution
315 	 *   <li>tileX, tileY: the tile x and y
316 	 * </ul>
317 	 * @public
318 	 * @name select
319 	 * @methodOf nanomaps.TileSelector.prototype
320 	 * @param {Projection} projection map projection
321 	 * @param {Number} resolution resolution at which tiles will be rendered
322 	 * @param {Number} x Global pixel coordinate relative to projection and resolution
323 	 * of the upper left corner of the region to select
324 	 * @param {Number} y Global pixel coordinate relative to projection and resolution
325 	 * of the upper left corner of the region to select
326 	 * @param {Number} width Pixel width of region to select
327 	 * @param {Number} height Pixel height of the region to select
328 	 * @param {Boolean} sort If true, then returns the tiles sorted by most visually impactful
329 	 * @return {Array[TileDesc]}
330 	 */
331 	select: function(projection, resolution, x, y, width, height, sort) {
332 		var tileStartX, tileStartY,
333 			tileEndX, tileEndY,
334 			tileMidX, tileMidY,
335 			i, j, tileDesc,
336 			nativeLevel=Math.round(projection.toLevel(resolution)),
337 			nativeResolution=projection.fromLevel(nativeLevel),
338 			unscaledOrigin=this.origin(projection),
339 			tileSize=this.tileSize,
340 			nativeOriginX=unscaledOrigin[0]/nativeResolution,
341 			nativeOriginY=unscaledOrigin[1]/nativeResolution,
342 			nativeScaleFactor=resolution / nativeResolution,
343 			retList=[];
344 
345 		// Scale x,y,width,height to our nativeResolution
346 		x=x*nativeScaleFactor;
347 		y=y*nativeScaleFactor;
348 		width=width*nativeScaleFactor;
349 		height=height*nativeScaleFactor;
350 		
351 		// Find the grid of tiles that intersect
352 		tileStartX=Math.floor( (x - nativeOriginX) / tileSize );
353 		tileStartY=Math.floor( (nativeOriginY - y) / tileSize );		// y-axis inversion
354 		tileEndX=Math.floor( ((x+width) - nativeOriginX ) / tileSize );
355 		tileEndY=Math.floor( (nativeOriginY - (y-height)) / tileSize );	// y-axis inversion
356 		//debugger;
357 		// Loop and report each one
358 		for (j=tileStartY; j<=tileEndY; j++) {
359 			for (i=tileStartX; i<=tileEndX; i++) {
360 				tileDesc={
361 					key: nativeLevel + ':' + i + ':' + j,
362 					res: nativeResolution,
363 					level: nativeLevel,
364 					tileX: i,
365 					tileY: j,
366 					size: tileSize,
367 					x: nativeOriginX + i*tileSize,
368 					y: nativeOriginY - j*tileSize	// y-axis inversion
369 				};
370 				
371 				retList.push(tileDesc);
372 			}
373 		}
374 		
375 		// Sort by proximity to center tile
376 		if (sort) {
377 			tileMidX=(tileStartX+tileEndX)/2;
378 			tileMidY=(tileStartY+tileEndY)/2;
379 			retList.sort(function(a, b) {
380 				var xa=Math.abs(a.tileX-tileMidX),
381 					ya=Math.abs(a.tileY-tileMidY),
382 					weighta=xa*xa+ya*ya,
383 					xb=Math.abs(b.tileX-tileMidX),
384 					yb=Math.abs(b.tileY-tileMidY),
385 					weightb=xb*xb+yb*yb;
386 				return weighta-weightb;
387 			});
388 		}
389 		
390 		return retList;
391 	}
392 };
