 17 function TileCache() {
 18 	var contents={},
 19 		marked=[];
 20 	
 21 	function disposeTileDesc(tileDesc) {
 22 		if (tileDesc.img&&tileDesc.img.parentNode) 
 23 			// Detach
 24 			tileDesc.img.parentNode.removeChild(tileDesc.img);
 25 	}
 26 		
 27 	this.mark=function() {
 28 		marked.length=0;
 29 	};
 30 	
 31 	this.free=function() {
 32 		var newContents={},
 33 			key, i, tileDesc;
 34 			
 35 		// Add all marked to the new dictionary
 36 		for (i=0; i<marked.length; i++) {
 37 			tileDesc=marked[i];
 38 			key=tileDesc.key;
 39 			newContents[key]=tileDesc;
 40 			delete contents[key];
 41 		}
 42 		
 43 		// Any existing in contents are to be discarded
 44 		for (i in contents) {
 45 			tileDesc=contents[i];
 46 			disposeTileDesc(tileDesc);
 47 		}
 48 		
 49 		// Swap
 50 		contents=newContents;
 51 		marked.length=0;
 52 	};
 53 	
 54 	this.use=function(tileDesc) {
 55 		var key=tileDesc.key,
 56 			existing=contents[key];
 57 		if (existing) {
 58 			disposeTileDesc(tileDesc);
 59 			tileDesc=existing;
 60 		}
 61 		contents[key]=tileDesc;
 62 		if (marked) marked.push(tileDesc);
 63 		return tileDesc;
 64 	};
 65 	
 66 	this.take=function(tileDesc) {
 67 		var key=tileDesc.key,
 68 			existing=contents[key];
 69 		if (existing) {
 70 			delete contents[key];
 71 			return existing;
 72 		}
 73 		return tileDesc;
 74 	};
 75 	
 76 	this.each=function(callback) {
 77 		var i, tileDesc;
 78 		for (i in contents) {
 79 			tileDesc=contents[i];
 80 			if (tileDesc && typeof tileDesc==='object')
 81 				callback(tileDesc);
 82 		}
 83 	};
 84 	
 85 	this.clear=function() {
 86 		contents={};
 87 		marked.length=0;
 88 	};
 89 	
 90 	this.moveFrom=function(otherCache) {
 91 		otherCache.each(function(tileDesc) {
 92 			var key=tileDesc.key;
 93 			if (contents[key]) {
 94 				// Already exists.  Delete
 95 				disposeTileDesc(tileDesc);
 96 			} else
 97 				contents[tileDesc.key]=tileDesc;
 98 		});
 99 		otherCache.clear();
100 	};
101 }

103 function TileLayer(options) {
104 	this.options=options=options||{};
105 	this.sel=new TileSelector(options);
106 	this.fgCache=new TileCache();
107 	this.bgCache=new TileCache();
108 }
109 TileLayer.prototype={
110 	createElement: function(map) {
111 		var element=document.createElement('div');
112 		element.style.position='absolute';
113 		element.style.left='0px';
114 		element.style.top='0px';
115 		element.mapDelegate=this;
116 		return element;
117 	},
118 	
119 	placeTile: function(map, element, tileDesc, moveToFront) {
120 		var img=tileDesc.img,
121 			transform=map.transform,
122 			zpx=transform.zpx,
123 			scaleFactor=tileDesc.res / transform.res;
124 			
125 		if (img&&img._error) {
126 			// Zero it out
127 			if (img.parentElement) img.parentElement.removeChild(img);
128 			tileDesc.img=null;
129 			img=null;
130 		}
131 		
132 		// If no valid img, instantiate it
133 		if (!img) {
134 			img=map.createElement('img');
135 			img.style.visibility='hidden';
136 			img.onload=makeVisibleOnLoad;
137 			img.style.position='absolute';
138 			img.src=this.sel.resolveSrc(tileDesc);
139 			tileDesc.img=img;
140 		}
141 		
142 		// Set position and size
143 		img.width=Math.ceil(tileDesc.size*scaleFactor);
144 		img.height=Math.ceil(tileDesc.size*scaleFactor);
145 		img.style.left=Math.round(tileDesc.x*scaleFactor - zpx[0]) + 'px';
146 		img.style.top=Math.round(zpx[1] - tileDesc.y*scaleFactor) + 'px';	// y-axis inversion
147 		if (moveToFront || img.parentNode!==element) element.appendChild(img);
148 	},
149 	
150 	onreset: function(map, element) {
151 		var self=this,
152 			transform=map.transform,
153 			buffer=self.options.buffer||64,
154 			ulXY=map.toGlobalPixels(-buffer,-buffer),
155 			width=map.width+buffer,
156 			height=map.height+buffer,
157 			curResolution=self.curResolution,
158 			displayResolution=transform.res,
159 			tileList=self.sel.select(transform.prj, displayResolution, ulXY.x, ulXY.y, width, height, true),
160 			fgCache=self.fgCache,
161 			bgCache=self.bgCache,
162 			refreshBackground=false,
163 			i,
164 			tileDesc;
165 			
166 		if (curResolution&&curResolution!==displayResolution) {
167 			// We have a scale change.  Snapshot the current tile
168 			// cache as the background cache and start on a new one
169 			bgCache.mark();
170 			bgCache.free();
171 			bgCache.moveFrom(fgCache);
172 			refreshBackground=true;
173 		}
174 
175 		fgCache.mark();
176 		for (i=0; i<tileList.length; i++) {
177 			tileDesc=fgCache.use(bgCache.take(tileList[i]));
178 			self.placeTile(map, element, tileDesc, true);
179 		}
180 		fgCache.free();
181 
182 		// Record so that the next time through we can tell whether
183 		// this is a simple change
184 		this.curResolution=displayResolution;
185 		
186 		if (refreshBackground) {
187 			bgCache.each(function(tileDesc) {
188 				self.placeTile(map, element, tileDesc);
189 			});
190 		}
191 		
192 	},
193 	
194 	onposition: function(map, element) {
195 		this.onreset(map, element);
196 	}
197 };

199 /**
200  * Class representing a tile selector for the layout of OSM, MS, Google, et al.
201  * Options:
202  *
203  */
204 function TileSelector(options) {
205 	options=options||{};
206 	// set defaults
207 	this.tileSize=options.tileSize||256;
208 	this.srcSpec="http://otile${modulo:1,2}.mqcdn.com/tiles/1.0.0/osm/${level}/${tileX}/${tileY}.png";
209 }
210 TileSelector.prototype={
211 	/**
212 	 * Gets the origin in physical units [x, y] for the tile coordinate space
213 	 */
214 	origin: function(projection) {
215 		var ret=this._origin;
216 		if (!ret) {
217 			ret=projection.forward(-180, 85.05112878);
218 			this._origin=ret;
219 		}
220 		return ret;
221 	},
222 	
223 	/**
224 	 * Get an img src from the tileDesc
225 	 */
226 	resolveSrc: function(tileDesc) {
227 		return this.srcSpec.replace(/\$\{([A-Za-z]+)(\:([^\}]*))?\}/g, function(s, name, ignore, args) {
228 			if (name==='modulo') {
229 				// get the args and modulo it by the tileDesc.tileX
230 				args=args.split(/\,/);
231 				return args[tileDesc.tileX%args.length];
232 			} else {
233 				return tileDesc[name];
234 			}
235 		});
236 	},
237 	
238 	/**
239 	 * Select all tiles that intersect the given bounding box at the
240 	 * given resolution.  Return an array of TileDesc elements, where
241 	 * each TileDesc has the following attributes:
242 	 *   - key: an opaque string that uniquely identifies this tile within
243 	 *     this instance where multiple requests for the same physical tile
244 	 *     have the same key
245 	 *   - res: the native resolution of the tile
246 	 *	 - level: the native zoom level
247 	 *   - x, y: origin at native resolution,
248 	 *	 - size: tile size (width/height) at native resolution
249 	 *   - tileX, tileY: the tile x and y
250 	 */
251 	select: function(projection, resolution, x, y, width, height, sort) {
252 		var tileStartX, tileStartY,
253 			tileEndX, tileEndY,
254 			tileMidX, tileMidY,
255 			i, j, tileDesc,
256 			nativeLevel=Math.round(projection.toLevel(resolution)),
257 			nativeResolution=projection.fromLevel(nativeLevel),
258 			unscaledOrigin=this.origin(projection),
259 			tileSize=this.tileSize,
260 			nativeOriginX=unscaledOrigin[0]/nativeResolution,
261 			nativeOriginY=unscaledOrigin[1]/nativeResolution,
262 			nativeScaleFactor=resolution / nativeResolution,
263 			retList=[];
264 
265 		// Scale x,y,width,height to our nativeResolution
266 		x=x*nativeScaleFactor;
267 		y=y*nativeScaleFactor;
268 		width=width*nativeScaleFactor;
269 		height=height*nativeScaleFactor;
270 		
271 		// Find the grid of tiles that intersect
272 		tileStartX=Math.floor( (x - nativeOriginX) / tileSize );
273 		tileStartY=Math.floor( (nativeOriginY - y) / tileSize );		// y-axis inversion
274 		tileEndX=Math.floor( ((x+width) - nativeOriginX ) / tileSize );
275 		tileEndY=Math.floor( (nativeOriginY - (y-height)) / tileSize );	// y-axis inversion
276 		//debugger;
277 		// Loop and report each one
278 		for (j=tileStartY; j<=tileEndY; j++) {
279 			for (i=tileStartX; i<=tileEndX; i++) {
280 				tileDesc={
281 					key: nativeLevel + ':' + i + ':' + j,
282 					res: nativeResolution,
283 					level: nativeLevel,
284 					tileX: i,
285 					tileY: j,
286 					size: tileSize,
287 					x: nativeOriginX + i*tileSize,
288 					y: nativeOriginY - j*tileSize	// y-axis inversion
289 				};
290 				
291 				retList.push(tileDesc);
292 			}
293 		}
294 		
295 		// Sort by proximity to center tile
296 		if (sort) {
297 			tileMidX=(tileStartX+tileEndX)/2;
298 			tileMidY=(tileStartY+tileEndY)/2;
299 			retList.sort(function(a, b) {
300 				var xa=Math.abs(a.tileX-tileMidX),
301 					ya=Math.abs(a.tileY-tileMidY),
302 					weighta=xa*xa+ya*ya,
303 					xb=Math.abs(b.tileX-tileMidX),
304 					yb=Math.abs(b.tileY-tileMidY),
305 					weightb=xb*xb+yb*yb;
306 				return weighta-weightb;
307 			});
308 		}
309 		
310 		return retList;
311 	}
312 };
