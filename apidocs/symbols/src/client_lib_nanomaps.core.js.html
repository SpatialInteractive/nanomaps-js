 27  * @private

 31 EventEmitterMethods._evt=EventEmitterMethods.listeners=function(name) {
 32 	var hash=this.__evt, list;
 33 	if (!hash) {
 34 		this.__evt=hash={};
 35 	}
 36 	list=hash[name];
 37 	if (!list) list=hash[name]=[];
 38 	return list;

108 /**
109  * Invokes a given event by name, triggering all registered persistent listeners
110  * and once only listeners.  All arguments after the first are taken to be the
111  * arguments to the event listeners.  Once only listeners are invoked first
112  * followed by persistent listeners.  Otherwise, listeners are invoked in the
113  * order added.
114  * <p>
115  * If a method exists on the instance named 'on' + event, then that method
116  * will be invoked as an event listener prior to any others.
117  *
118  * <p><i>TODO: Make listener add/remove durable from within a callback</i>
119  *
120  * @public
121  * @methodOf nanomaps.EventEmitter.prototype
122  * @name emit
123  * @param {string} event Event name to add listener to
124  * @param ... Arguments to registred listeners 
125  */
126 EventEmitterMethods.emit=function(event /*, arg1..argn */) {
127 	var i, list=this._evt(event), eventArgs=Array.prototype.slice.call(arguments, 1),
128 		handler=this['on' + event];
129 	
130 	// Emit on this object
131 	if (typeof handler==='function') {
132 		handler.apply(this, eventArgs);
133 	}
134 	
135 	// Emit once events
136 	list=this._evt(event+'$once');
137 	for (i=0; i<list.length; i++) {
138 		list[i].apply(this, eventArgs);
139 	}
140 	list.length=0;	// Zero the once only array
141 
142 	// Emit standard events
143 	for (i=0; i<list.length; i++) {
144 		list[i].apply(this, eventArgs);
145 	}
146 };

148 /**
149  * Overrides the given methodName on this instance applying the given
150  * advice ("before" or "after").  The given target method is invoked
151  * for the advice.
152  * 
153  * <p>Invoking this method supports the arbitrary augmentation of a class's
154  * methods with before or after advice.  The original method is replaced with
155  * a stub that invokes a list of other methods before and after the original
156  * target.  If the original method doesn't exist, the stub will still be
157  * defined.
158  * <p>It is safe to use this method on prototypes or instances as it will only
159  * modify the outermost prototype (target of the call).
160  *
161  * @example MapSurface.prototype.advise('initialize', 'after', 
162  *		function() { alert('Initialized'); });
163  * @example myMap.advise('onzoom', 'after', 
164  *		function() { alert('zoomed'); });
165  * @public
166  * @methodOf nanomaps.EventEmitter.prototype
167  * @name advise
168  * @param {string} methodName Name of the method to override
169  * @param {string} advice Advice to apply ("before" or "after")
170  * @param {function} target Function which is the target of advice
171  */
172 EventEmitterMethods.advise=function(methodName, advice, target) {
173 	var originalMethod=this[methodName],
174 		advisorStub, adviceList;
175 		
176 	if (!originalMethod || !originalMethod.__stub__ || !this.hasOwnProperty(methodName))
177 		advisorStub=this[methodName]=getAdvisorStub(originalMethod);
178 	else
179 		advisorStub=originalMethod;
180 	
181 	adviceList=advisorStub[advice];
182 	if (adviceList) adviceList.push(target);
183 };
184 
185 /**
186  * @private
187  */
188 function getAdvisorStub(method) {
189 	var stub=function() {
190 		var i, thisFunction=arguments.callee,
191 			list, retValue;
192 		
193 		// Apply before advice
194 		list=thisFunction.before;
195 		for (i=0; i<list.length; i++) {
196 			list[i].apply(this, arguments);
197 		}
198 		
199 		// Invoke original method
200 		if (method) retValue=method.apply(this, arguments);
201 		
202 		// Invoke after method
203 		list=thisFunction.after;
204 		for (i=0; i<list.length; i++) {
205 			list[i].apply(this, arguments);
206 		}
207 	};
208 	stub.__stub__=true;
209 	stub.before=[];
210 	stub.after=[];
211 	return stub;
212 }

216  * and routes the event to the target instance, calling its emit method with

229 /**
230  * Initialize an existing DOM element as a map.  All internal map structures
231  * are added before any existing content in the element.
232  *
233  * <h2>Sizing</h2>
234  * The map size must be explicitly maintained.  If not specified, then the natural
235  * size of the containing element is used.  If this natural size ever changes,
236  * setSize() must be called to reset it.
237  * 
238  * @constructor
239  * @extends nanomaps.EventEmitter
240  * @name nanomaps.MapSurface
241  * @param {HTMLElement} elt The map container element
242  * @param {integer} [options.width] Explicit width of the map
243  * @param {integer} [options.height] Explicit height of the map
244  * @param {number} [options.resolution] Initial resolution
245  * @param {Projection} [options.projection=new nanomaps.WebMercatorProjection()] Map projection 
246  * @param {LatLngObject} [options.center] Initial center of the map
247  */
248 function MapSurface(elt, options) {
249 	if (!options) options={};
250 	var document=options.document||elt.ownerDocument||window.document,
251 		width=options.width, height=options.height, attr,
252 		viewportElt, glassElt, managedElt, center, projection;
253 		
254 	// Local functions
255 	function createElement(name) {
256 		return document.createElement(name);
257 	}
258 	
259 	/**
260 	 * Create an element in the document that this map exists in.
261 	 * @methodOf nanomaps.MapSurface#
262 	 * @name createElement
263 	 * @return {HTMLElement}
264 	 */
265 	this.createElement=createElement;
266 	
267 	// Hardcode some important styles
268 	elt.style.overflow='visible';
269 	attr=elt.style.position;
270 	if (attr!='relative' && attr!='absolute' && attr!='fixed')
271 		elt.style.position='relative';	// Make positioned
272 	attr=elt.style.zIndex;
273 	if (attr==='' || attr==='auto')
274 		elt.style.zIndex='inherit';	// Establish a new stacking context
275 	
276 	function createOverlay() {
277 		var overlay=createElement('div');
278 		overlay.style.overflow='hidden';
279 		overlay.style.position='absolute';
280 		overlay.style.left='0px';
281 		overlay.style.top='0px';
282 		overlay.style.width='100%';
283 		overlay.style.height='100%';
284 		return overlay;
285 	}
286 	
287 	// create glass
288 	glassElt=createOverlay();
289 	glassElt.className='glass';
290 	glassElt.style.display='block';
291 	elt.insertBefore(glassElt, elt.firstChild);
292 	
293 	
294 	// create viewport
295 	viewportElt=createOverlay();
296 	viewportElt.className='viewport';
297 	elt.insertBefore(viewportElt, elt.firstChild);
298 	
299 	// create manage
300 	managedElt=createElement('div');
301 	managedElt.className='managed';
302 	managedElt.style.position='absolute';
303 	managedElt.style.width='100%';
304 	managedElt.style.height='100%';
305 	viewportElt.appendChild(managedElt);
306 	
307 	// Dictionary of elements
308 	this.elements={
309 		document: document,
310 		managed: managedElt,
311 		viewport: viewportElt,
312 		glass: glassElt,
313 		parent: elt
314 	};
315 	
316 	// Fixed size or autosize
317 	if (typeof width!=='number' || typeof height!=='number') {
318 		this.setSize();
319 		// Auto width/height (get inner size)
320 		width=elt.clientWidth;
321 		height=elt.clientHeight;
322 	} else {
323 		this.setSize(width, height);
324 	}
325 	
326 	// Initialize transform
327 	projection=options.projection||new Projections.WebMercator();
328 	center=options.center||projection.DEFAULT_CENTER;
329 	this.transform=new MapTransform();
330 	this.transform.init(
331 		projection,
332 		options.resolution||projection.DEFAULT_RESOLUTION,
333 		[center.lng, center.lat]
334 		);
335 	
336 	// Setup initial state by setting center
337 	this.setCenter(center);
338 	
339 	// Initialization hook
340 	this.initialize(options);
341 	
342 	// Collect loose content
343 	this.collect();
344 }

346 /**
347  * Iterate over each child element of the global layer, invoking
348  * callback.  The this reference is preserved as this instance in
349  * calls.
350  * @private
351  * @methodOf nanomaps.MapSurface.prototype
352  * @name _each
353  */
354 MapSurfaceMethods._each=function(includeManaged, callback) {
355 	var elements=this.elements, managed=elements.managed, viewport=elements.viewport;
356 	for (var childElt=viewport.firstChild; childElt; childElt=childElt.nextSibling) {
357 		if (childElt.nodeType!==1 || childElt===managed) continue;	// Skip non-elements
358 		callback.call(this, childElt);
359 	}
360 	if (includeManaged) {
361 		for (childElt=managed.firstChild; childElt; childElt=childElt.nextSibling) {
362 			if (childElt.nodeType!==1) continue;	// Skip non-elements
363 			callback.call(this, childElt);
364 		}
365 	}
366 };

470 /**
471  * Adds content to the map.  This is the primary method of attaching DOM content
472  * to the map after construction.  Give it either an HTMLElement or a factory
473  * object that supports a createElement(mapSurface) method.
474  * 
475  * <h2>Display Management</h2>
476  * The primary thing that map content needs to be able to do is position itself
477  * relative to a global coordinate system.  Each element managed by the map
478  * viewport is checked for a special "mapDelegate" property.  If defined, then
479  * this delegate object provides callbacks that will be invoked on map geometry
480  * changes and are responsible for displaying the element properly.  See the
481  * documentation for a complete treatment of MapDelegate objects.
482  *
483  * <h2>Default Behavior</h2>
484  * If an element added to the map does not have a mapDelegate property, the
485  * default delegate is used instead.  This default delegate defines standard
486  * and intuitive behavior for positioning a simple object.  The position and
487  * pixel offset are determined by taking the first defined result from the
488  * following sequence:
489  * <ol>
490  * <li>If the "geo" property exists on the DOM object.  Take the "latitude", 
491  *		"longitude", "x" and "y" properties.
492  * <li>Read the "latitude" and "longitude" attributes of the element.
493  * </ol>
494  *
495  * @public
496  * @methodOf nanomaps.MapSurface.prototype
497  * @name attach
498  * @param {HTMLElement or factory} element Element or factory to add
499  */
500 MapSurfaceMethods.attach=function(element) {
501 	// Detect if HTMLElement in standards compliant way with fallbacks for IE
502 	var isHTML=window.HTMLElement ? element instanceof HTMLElement : (element.nodeType===1);
503 	if (!isHTML) {
504 		// Treat as a delegate
505 		element=element.createElement(this);
506 	}
507 	
508 	var delegate=element.mapDelegate||DEFAULT_MAP_DELEGATE,
509 		elements=this.elements;
510 
511 	element.style.position='absolute';	// Make positioned
512 	if (delegate.unmanaged) {
513 		// Add to the viewport before the global element
514 		elements.viewport.insertBefore(element, elements.managed);
515 	} else {
516 		// Add to the managed element
517 		elements.managed.appendChild(element);
518 	}
519 	
520 	this._notifyResetSingle(element);
521 };

597 MapSurfaceMethods.setResolution=function(resolution, preserveXY) {
598 	var center=this._center, deltaX, deltaY, centerLngLat;
599 	
600 	//if (preserveXY) console.log('setResolution(' + resolution + ', preserveX=' + preserveXY.x + ', preserveY=' + preserveXY.y + ')');
601 	if (preserveXY) {
602 		// re-interpret the center such that the point at the given
603 		// offset in the viewport stays at the given point
604 		//console.log('original center: lat=' + center.lat + ', lng=' + center.lng);
605 		center=this.toLatLng(preserveXY.x, preserveXY.y);
606 		//console.log('revised center: lat=' + center.lat + ', lng=' + center.lng);
607 		deltaX=this.width/2-preserveXY.x;
608 		deltaY=this.height/2-preserveXY.y;
609 		// deltaX and deltaY = offset from center to recenter at
610 	}
611 	
612 	this.transform=this.transform.rescale(resolution, [center.lng, center.lat]);
613 	
614 	if (preserveXY) {
615 		// Reset the center based on the offset
616 		// Can't use this.toLatLng here because we are not yet in a consistent
617 		// state (_updateCenter has not been called) and toLatLng uses this to
618 		// do viewport relative coordinates.  Take advantage of the fact that
619 		// we know the zero pixel point to be the lat/lng under the original
620 		// prserveXY coordinates to avoid the need for viewport biasing.
621 		//console.log('Biasing center by (' + deltaX + ',' + deltaY + ')px');
622 		centerLngLat=this.transform.fromSurface(deltaX, deltaY);
623 		if (centerLngLat) {
624 			center.lng=centerLngLat[0];
625 			center.lat=centerLngLat[1];
626 		}
627 	}
628 	
629 	this._updateCenter(center);
630 	this._notifyReset();
631 };

649 /**
650  * Given x,y coordinates relative to the visible area of the viewport,
651  * return the corresponding lat/lng
652  */
653 MapSurfaceMethods.toLatLng=function(x, y) {
654 	var transform=this.transform, managed=this.elements.managed, lngLat;
655 	
656 	//console.log('toLatLng(' + x + ',' + y + ')');
657 	//console.log('global left=' + global.style.left + ', top=' + global.style.top);
658 	
659 	x-=parseInt(managed.style.left);
660 	y-=parseInt(managed.style.top);
661 	
662 	//console.log('zpx rel xy=(' + x + ',' + y + ')');
663 	
664 	lngLat=transform.fromSurface(x, y);
665 	if (!lngLat) return null;
666 	return {lng: lngLat[0], lat: lngLat[1]};
667 };

671  * managed pixel coordinates at the current resolution.
672  */
673 MapSurfaceMethods.toGlobalPixels=function(x, y) {
674 	var transform=this.transform, managed=this.elements.managed;

676 	y-=parseInt(managed.style.top);
677 	return {

697 function extractDefaultPosition(element) {
698 	var geo=element.geo;
699 	if (!geo) {
700 		geo={};
701 		if (isNaN(geo.latitude=Number(element.getAttribute('latitude')||'NaN'))) return null;
702 		if (isNaN(geo.longitude=Number(element.getAttribute('longitude')||'NaN'))) return null;
703 		geo.xoffset=Number(element.getAttribute('xoffset')||'NaN');
704 		geo.yoffset=Number(element.getAttribute('yoffset')||'NaN');
705 	}
706 	return geo;

736 /**
737  * Construct a new MapTransform with the given parameters:
738  *  - projection: The projection object for the surface
739  *  - resolution: Ground resolution in m/px
740  *  - zeroLat: The latitude corresponding with the top of the viewport
741  *  - zeroLng: The longitude corresponding with the left of the viewport
742  */
743 function MapTransform() {
744 	this.sequence=__nextId++;
745 }
746 MapTransform.prototype={
747 	init: function(projection, resolution, zeroLngLat) {
748 		// The sequence number is used to detect if objects are
749 		// aligned properly against the current transform
750 		this.prj=projection;
751 		this.res=resolution;
752 		this.zll=zeroLngLat;
753 		
754 		// Calculate corresponding unaligned pixel coordinates associated
755 		// with zeroLatLng
756 		this.zpx=this.toPixels(zeroLngLat[0], zeroLngLat[1]);
757 	},
758 	
759 	/**
760 	 * Return a new MapTransform at a new scale and zeroLatLng
761 	 */
762 	rescale: function(resolution, zeroLngLat) {
763 		var transform=new MapTransform();
764 		transform.init(this.prj, resolution, zeroLngLat);
765 		return transform;
766 	},
767 	
768 	/**
769 	 * Convert the given longitude/latitude to returned [x, y] coordinates.
770 	 * Returns null if the lng/lat is out of bounds.
771 	 * NOTE: the order of the parameters is longitude, latitude, corresponding
772 	 * with x and y.
773 	 * In order to align to the viewport, the return value should be subtracted
774 	 * from this.zeroPx (see toViewport).
775 	 */
776 	toPixels: function(lng, lat) {
777 		var xy=this.prj.forward(lng, lat), resolution=this.res;
778 		if (!xy) return null;
779 		xy[0]/=resolution;
780 		xy[1]/=resolution;
781 		
782 		return xy;
783 	},
784 	
785 	/**
786 	 * Return surface coordinates of the given lat/lng
787 	 */
788 	toSurface: function(lng, lat) {
789 		var xy=this.toPixels(lng, lat);
790 		if (!xy) return null;
791 		xy[0]-=this.zpx[0];
792 		xy[1]=this.zpx[1] - xy[1]; // Note Y axis inversion
793 		return xy;
794 	},
795 	
796 	/**
797 	 * Convert from global pixel coordinates to lng/lat.
798 	 */
799 	fromPixels: function(x, y) {
800 		var resolution=this.res;
801 		return this.prj.inverse(x * resolution, y * resolution);
802 	},
803 	
804 	/**
805 	 * Convert from surface coordinates to [lng, lat]
806 	 */
807 	fromSurface: function(x, y) {
808 		return this.fromPixels(x+this.zpx[0], this.zpx[1] - y);	// Note Y axis inversion
809 	}
810 };

813 var Projections={
814 	/**
815 	 * Standard "Web Mercator" projection as used by Google, Microsoft, OSM, et al.
816 	 * Instantiate as new WebMercator(options).  Options can include:
817 	 *   - MIN_LEVEL: Minimum zoom level.  Default==1.
818 	 *   - MAX_LEVEL: Maximum zoom level.  Default==18.
819 	 */
820 	WebMercator: function(options) {
821 		if (!options) options={};
822 		this.DEFAULT_CENTER={lat:39.7406, lng:-104.985441};
823 		this.DEFAULT_RESOLUTION=611.4962;
824 	
825 		var EARTH_RADIUS=6378137.0,
826 			DEG_TO_RAD=.0174532925199432958,
827 			RAD_TO_DEG=57.29577951308232,
828 			FOURTHPI=0.78539816339744833,
829 			HALFPI=1.5707963267948966,
830 			HIGHEST_RES=78271.5170;
831 		
832 		this.MIN_LEVEL=options.MIN_LEVEL||1;
833 		this.MAX_LEVEL=options.MAX_LEVEL||18;
834 		
835 		this.forward=function(x, y) {
836 			return [
837 				x*DEG_TO_RAD * EARTH_RADIUS, 
838 				Math.log(Math.tan(FOURTHPI + 0.5 * DEG_TO_RAD * y)) * EARTH_RADIUS 
839 			];
840 		};
841 		
842 		this.inverse=function(x, y) {
843 			return [
844 				RAD_TO_DEG * x / EARTH_RADIUS,
845 				RAD_TO_DEG * (HALFPI - 2. * Math.atan(Math.exp(-y/EARTH_RADIUS)))
846 			];
847 		};
848 		
849 		/**
850 		 * Return the resolution (m/px) for the given zoom index given a standard
851 		 * power of two zoom breakdown.
852 		 */
853 		this.fromLevel=function(level) {
854 			return HIGHEST_RES/Math.pow(2, level-1);
855 		}
856 		
857 		this.toLevel=function(resolution) {
858 			return Math.log(HIGHEST_RES/resolution) / Math.log(2) + 1;
859 		}
860 		
861 		// release memory from closure
862 		options=null;
863 	}
864 };
