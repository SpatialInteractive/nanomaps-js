108 /**
109  * Invokes a given event by name, triggering all registered persistent listeners
110  * and once only listeners.  All arguments after the first are taken to be the
111  * arguments to the event listeners.  Once only listeners are invoked first
112  * followed by persistent listeners.  Otherwise, listeners are invoked in the
113  * order added.
114  * <p>
115  * If a method exists on the instance named 'on' + event, then that method
116  * will be invoked as an event listener prior to any others.
117  *
118  * <p><i>TODO: Make listener add/remove durable from within a callback</i>
119  *
120  * @public
121  * @methodOf nanomaps.EventEmitter.prototype
122  * @name emit
123  * @param {string} event Event name to add listener to
124  * @param ... Arguments to registred listeners 
125  */
126 EventEmitterMethods.emit=function(event /*, arg1..argn */) {
127 	var i, list=this._evt(event), eventArgs=Array.prototype.slice.call(arguments, 1),
128 		handler=this['on' + event];
129 	
130 	// Emit on this object
131 	if (typeof handler==='function') {
132 		handler.apply(this, eventArgs);
133 	}
134 	
135 	// Emit once events
136 	list=this._evt(event+'$once');
137 	for (i=0; i<list.length; i++) {
138 		list[i].apply(this, eventArgs);
139 	}
140 	list.length=0;	// Zero the once only array
141 
142 	// Emit standard events
143 	for (i=0; i<list.length; i++) {
144 		list[i].apply(this, eventArgs);
145 	}
146 };

148 /**
149  * Overrides the given methodName on this instance applying the given
150  * advice ("before" or "after").  The given target method is invoked
151  * for the advice.
152  * 
153  * <p>Invoking this method supports the arbitrary augmentation of a class's
154  * methods with before or after advice.  The original method is replaced with
155  * a stub that invokes a list of other methods before and after the original
156  * target.  If the original method doesn't exist, the stub will still be
157  * defined.
158  * <p>It is safe to use this method on prototypes or instances as it will only
159  * modify the outermost prototype (target of the call).
160  *
161  * @example MapSurface.prototype.advise('initialize', 'after', 
162  *		function() { alert('Initialized'); });
163  * @example myMap.advise('onzoom', 'after', 
164  *		function() { alert('zoomed'); });
165  * @public
166  * @methodOf nanomaps.EventEmitter.prototype
167  * @name advise
168  * @param {string} methodName Name of the method to override
169  * @param {string} advice Advice to apply ("before" or "after")
170  * @param {function} target Function which is the target of advice
171  */
172 EventEmitterMethods.advise=function(methodName, advice, target) {
173 	var originalMethod=this[methodName],
174 		advisorStub, adviceList;
175 		
176 	if (!originalMethod || !originalMethod.__stub__ || !this.hasOwnProperty(methodName))
177 		advisorStub=this[methodName]=getAdvisorStub(originalMethod);
178 	else
179 		advisorStub=originalMethod;
180 	
181 	adviceList=advisorStub[advice];
182 	if (adviceList) adviceList.push(target);
183 };
184 
185 /**
186  * @private
187  */
188 function getAdvisorStub(method) {
189 	var stub=function() {
190 		var i, thisFunction=arguments.callee,
191 			list, retValue;
192 		
193 		// Apply before advice
194 		list=thisFunction.before;
195 		for (i=0; i<list.length; i++) {
196 			list[i].apply(this, arguments);
197 		}
198 		
199 		// Invoke original method
200 		if (method) retValue=method.apply(this, arguments);
201 		
202 		// Invoke after method
203 		list=thisFunction.after;
204 		for (i=0; i<list.length; i++) {
205 			list[i].apply(this, arguments);
206 		}
207 	};
208 	stub.__stub__=true;
209 	stub.before=[];
210 	stub.after=[];
211 	return stub;
212 }

229 /**
230  * Initialize an existing DOM element as a map.  All internal map structures
231  * are added before any existing content in the element.
232  *
233  * <h2>Sizing</h2>
234  * The map size must be explicitly maintained.  If not specified, then the natural
235  * size of the containing element is used.  If this natural size ever changes,
236  * setSize() must be called to reset it.
237  * 
238  * @example
239  * var map=new nanomaps.MapSurface(someElement);
240  * map.attach(new nanomaps.TileLayer({ 
241  *    tileSrc: "http://otile${modulo:1,2,3}.mqcdn.com/tiles/1.0.0/osm/${level}/${tileX}/${tileY}.png" }));
242  * 
243  * @constructor
244  * @extends nanomaps.EventEmitter
245  * @name nanomaps.MapSurface
246  * @param {HTMLElement} elt The map container element
247  * @param {integer} [options.width] Explicit width of the map
248  * @param {integer} [options.height] Explicit height of the map
249  * @param {number} [options.resolution] Initial resolution
250  * @param {Projection} [options.projection=new nanomaps.WebMercatorProjection()] Map projection 
251  * @param {LatLngObject} [options.center] Initial center of the map
252  */
253 function MapSurface(elt, options) {
254 	if (!options) options={};
255 	var document=options.document||elt.ownerDocument||window.document,
256 		width=options.width, height=options.height, attr,
257 		viewportElt, glassElt, managedElt, center, projection;
258 		
259 	// Local functions
260 	function createElement(name) {
261 		return document.createElement(name);
262 	}
263 	
264 	/**
265 	 * Create an element in the document that this map exists in.
266 	 * @methodOf nanomaps.MapSurface#
267 	 * @name createElement
268 	 * @return {HTMLElement}
269 	 */
270 	this.createElement=createElement;
271 	
272 	// Hardcode some important styles
273 	elt.style.overflow='visible';
274 	attr=elt.style.position;
275 	if (attr!='relative' && attr!='absolute' && attr!='fixed')
276 		elt.style.position='relative';	// Make positioned
277 	attr=elt.style.zIndex;
278 	if (attr==='' || attr==='auto')
279 		elt.style.zIndex='inherit';	// Establish a new stacking context
280 	
281 	function createOverlay() {
282 		var overlay=createElement('div');
283 		overlay.style.overflow='hidden';
284 		overlay.style.position='absolute';
285 		overlay.style.left='0px';
286 		overlay.style.top='0px';
287 		overlay.style.width='100%';
288 		overlay.style.height='100%';
289 		return overlay;
290 	}
291 	
292 	// create glass
293 	glassElt=createOverlay();
294 	glassElt.className='glass';
295 	glassElt.style.display='block';
296 	elt.insertBefore(glassElt, elt.firstChild);
297 	
298 	
299 	// create viewport
300 	viewportElt=createOverlay();
301 	viewportElt.className='viewport';
302 	elt.insertBefore(viewportElt, elt.firstChild);
303 	
304 	// create manage
305 	managedElt=createElement('div');
306 	managedElt.className='managed';
307 	managedElt.style.position='absolute';
308 	managedElt.style.width='100%';
309 	managedElt.style.height='100%';
310 	viewportElt.appendChild(managedElt);
311 	
312 	// Dictionary of elements
313 	this.elements={
314 		document: document,
315 		managed: managedElt,
316 		viewport: viewportElt,
317 		glass: glassElt,
318 		parent: elt
319 	};
320 	
321 	// Fixed size or autosize
322 	if (typeof width!=='number' || typeof height!=='number') {
323 		this.setSize();
324 		// Auto width/height (get inner size)
325 		width=elt.clientWidth;
326 		height=elt.clientHeight;
327 	} else {
328 		this.setSize(width, height);
329 	}
330 	
331 	// Initialize transform
332 	projection=options.projection||new Projections.WebMercator();
333 	center=options.center||projection.DEFAULT_CENTER;
334 	this.transform=new MapTransform();
335 	this.transform.init(
336 		projection,
337 		options.resolution||projection.DEFAULT_RESOLUTION,
338 		[center.lng, center.lat]
339 		);
340 	
341 	// Setup initial state by setting center
342 	this.setCenter(center);
343 	
344 	// Initialization hook
345 	this.initialize(options);
346 	
347 	// Collect loose content
348 	this.collect();
349 }

351 /**
352  * Iterate over each child element of the global layer, invoking
353  * callback.  The this reference is preserved as this instance in
354  * calls.
355  * @private
356  * @methodOf nanomaps.MapSurface.prototype
357  * @name _each
358  */
359 MapSurfaceMethods._each=function(includeManaged, callback) {
360 	var elements=this.elements, managed=elements.managed, viewport=elements.viewport;
361 	for (var childElt=viewport.firstChild; childElt; childElt=childElt.nextSibling) {
362 		if (childElt.nodeType!==1 || childElt===managed) continue;	// Skip non-elements
363 		callback.call(this, childElt);
364 	}
365 	if (includeManaged) {
366 		for (childElt=managed.firstChild; childElt; childElt=childElt.nextSibling) {
367 			if (childElt.nodeType!==1) continue;	// Skip non-elements
368 			callback.call(this, childElt);
369 		}
370 	}
371 };

475 /**
476  * Adds content to the map.  This is the primary method of attaching DOM content
477  * to the map after construction.  Give it either an HTMLElement or a factory
478  * object that supports a createElement(mapSurface) method.
479  * 
480  * <h2>Display Management</h2>
481  * The primary thing that map content needs to be able to do is position itself
482  * relative to a global coordinate system.  Each element managed by the map
483  * viewport is checked for a special "mapDelegate" property.  If defined, then
484  * this delegate object provides callbacks that will be invoked on map geometry
485  * changes and are responsible for displaying the element properly.  See the
486  * documentation for a complete treatment of MapDelegate objects.
487  *
488  * <h2>Default Behavior</h2>
489  * If an element added to the map does not have a mapDelegate property, the
490  * default delegate is used instead.  This default delegate defines standard
491  * and intuitive behavior for positioning a simple object.  The position and
492  * pixel offset are determined by taking the first defined result from the
493  * following sequence:
494  * <ol>
495  * <li>If the "geo" property exists on the DOM object.  Take the "latitude", 
496  *		"longitude", "x" and "y" properties.
497  * <li>Read the "latitude" and "longitude" attributes of the element.
498  * </ol>
499  *
500  * @public
501  * @methodOf nanomaps.MapSurface.prototype
502  * @name attach
503  * @param {HTMLElement or factory} element Element or factory to add
504  * @return {HTMLElement} the element attached
505  */
506 MapSurfaceMethods.attach=function(element) {
507 	// Detect if HTMLElement in standards compliant way with fallbacks for IE
508 	var isHTML=window.HTMLElement ? element instanceof HTMLElement : (element.nodeType===1);
509 	if (!isHTML) {
510 		// Treat as a delegate
511 		element=element.createElement(this);
512 	}
513 	
514 	var delegate=element.mapDelegate||DEFAULT_MAP_DELEGATE,
515 		elements=this.elements;
516 
517 	element.style.position='absolute';	// Make positioned
518 	if (delegate.unmanaged) {
519 		// Add to the viewport before the global element
520 		elements.viewport.insertBefore(element, elements.managed);
521 	} else {
522 		// Add to the managed element
523 		elements.managed.appendChild(element);
524 	}
525 	
526 	this._notifyResetSingle(element);
527 	
528 	return element;
529 };

658 /**
659  * Change the map resolution, optionaly preserving the global coordinates
660  * under an arbitrary location on the map viewport.  If preserveXY is not
661  * given, then the coordinates at the center are preserved.
662  *
663  * @public
664  * @methodOf nanomaps.MapSurface.prototype
665  * @name setResolution
666  * @param {Number} resolution
667  * @param {XY object} [preserveXY] Object with "x" and "y" properties specifying
668  * 		a point on the viewport measured from the upper-left
669  */
670 MapSurfaceMethods.setResolution=function(resolution, preserveXY) {
671 	var center=this._center, deltaX, deltaY, centerLngLat;
672 	
673 	//if (preserveXY) console.log('setResolution(' + resolution + ', preserveX=' + preserveXY.x + ', preserveY=' + preserveXY.y + ')');
674 	if (preserveXY) {
675 		// re-interpret the center such that the point at the given
676 		// offset in the viewport stays at the given point
677 		//console.log('original center: lat=' + center.lat + ', lng=' + center.lng);
678 		center=this.toLatLng(preserveXY.x, preserveXY.y);
679 		//console.log('revised center: lat=' + center.lat + ', lng=' + center.lng);
680 		deltaX=this.width/2-preserveXY.x;
681 		deltaY=this.height/2-preserveXY.y;
682 		// deltaX and deltaY = offset from center to recenter at
683 	}
684 	
685 	this.transform=this.transform.rescale(resolution, [center.lng, center.lat]);
686 	
687 	if (preserveXY) {
688 		// Reset the center based on the offset
689 		// Can't use this.toLatLng here because we are not yet in a consistent
690 		// state (_updateCenter has not been called) and toLatLng uses this to
691 		// do viewport relative coordinates.  Take advantage of the fact that
692 		// we know the zero pixel point to be the lat/lng under the original
693 		// prserveXY coordinates to avoid the need for viewport biasing.
694 		//console.log('Biasing center by (' + deltaX + ',' + deltaY + ')px');
695 		centerLngLat=this.transform.fromSurface(deltaX, deltaY);
696 		if (centerLngLat) {
697 			center.lng=centerLngLat[0];
698 			center.lat=centerLngLat[1];
699 		}
700 	}
701 	
702 	this._updateCenter(center);
703 	this._notifyReset();
704 };

706 /**
707  * Change the map resolution by specifying a projection dependent level, 
708  * optionaly preserving the global coordinates
709  * under an arbitrary location on the map viewport.  If preserveXY is not
710  * given, then the coordinates at the center are preserved.
711  * <p>
712  * The mapping between level and resolution is done by the projection.  For the
713  * default WebMercator projection, the level is a positive number representing
714  * the power of two decimation factor.  It can take any floating point value
715  * (not just integers) and is interpolated using 2^x exponentiation or a base
716  * 2 logarithm.
717  * 
718  * @public
719  * @methodOf nanomaps.MapSurface.prototype
720  * @name setLevel
721  * @param {Number} level Floating point level between the projections minimum and
722  * 		maximum zoom level (if out of these bounds, it is clamped to min/max
723  *		values)
724  * @param {XY object} [preserveXY] Object with "x" and "y" properties specifying
725  * 		a point on the viewport measured from the upper-left
726  */
727 MapSurfaceMethods.setLevel=function(level, preserveXY) {
728 	var prj=this.transform.prj, minLevel=prj.MIN_LEVEL, maxLevel=prj.MAX_LEVEL;
729 	
730 	if (!level) return;
731 	if (level>maxLevel) level=maxLevel;
732 	else if (level<minLevel) level=minLevel;
733 	
734 	this.setResolution(prj.fromLevel(level), preserveXY);
735 };

754  * return the corresponding lat/lng.
755  *
756  * @public
757  * @methodOf nanomaps.MapSurface.prototype
758  * @name toLatLng

760  * @param {Number} y coordinate measured from the top of the viewport
761  * @return {LatLng object}
762  */
763 MapSurfaceMethods.toLatLng=function(x, y) {
764 	var transform=this.transform, managed=this.elements.managed, lngLat;
765 	
766 	//console.log('toLatLng(' + x + ',' + y + ')');
767 	//console.log('global left=' + global.style.left + ', top=' + global.style.top);
768 	
769 	x-=parseInt(managed.style.left);

779 /**
780  * Translate a viewport coordinate relative to the visible area to the
781  * projected pixel coordinates relative to the managed layer coordinate system.  
782  * Note that this assumes an axis inversion
783  * on the y-axis (ie. Increasing "y" parameters will produce decreasing "y"
784  * results).  This method is used for a number of internal calculations but
785  * is unlikely to be of use to end-callers.
786  *
787  * @public
788  * @methodOf nanomaps.MapSurface.prototype
789  * @name toGlobalPixels
790  * @param {Number} x coordinate measured from the left of the viewport
791  * @param {Number} y coordinate measured from the top of the viewport
792  * @return {XY object}
793  */
794 MapSurfaceMethods.toGlobalPixels=function(x, y) {
795 	var transform=this.transform, managed=this.elements.managed;
796 	x-=parseInt(managed.style.left);
797 	y-=parseInt(managed.style.top);
798 	return {
799 		x: x + transform.zpx[0],
800 		y: transform.zpx[1] - y
801 	};
802 };

867 /**
868  * Construct a new MapTransform with the given parameters.  Implementations
869  * will typically call init(...) next.  Note that this class typically represents
870  * longitude and latitude as ordered arrays as [longitude, latitude], corresponding
871  * to x and y.
872  * 
873  * @class
874  * @name nanomaps.MapTransform
875  * @public
876  */
877 function MapTransform() {
878 	this.sequence=__nextId++;
879 }
880 MapTransform.prototype=
881 /**
882  * @lends nanomaps.MapTransform.prototype
883  */
884 {
885 	/**
886 	 * @public
887 	 * @param {Projection object} projection map projection
888 	 * @param {Number} resolution the resolution that the transform describes
889 	 * @param {Array[lng,lat]} Longitude/Latitude corresponding to the viewport
890 	 * zero pixel coordinate
891 	 */
892 	init: function(projection, resolution, zeroLngLat) {
893 		// The sequence number is used to detect if objects are
894 		// aligned properly against the current transform
895 		this.prj=projection;
896 		this.res=resolution;
897 		this.zll=zeroLngLat;
898 		
899 		// Calculate corresponding unaligned pixel coordinates associated
900 		// with zeroLatLng
901 		this.zpx=this.toPixels(zeroLngLat[0], zeroLngLat[1]);
902 	},
903 	
904 	/**
905 	 * Return a new MapTransform at a new scale and zeroLatLng
906 	 * @public
907 	 * @param {Number} resolution new resolution
908 	 * @param {Array[lng,lat]} zeroLngLat new zero point
909 	 */
910 	rescale: function(resolution, zeroLngLat) {
911 		var transform=new MapTransform();
912 		transform.init(this.prj, resolution, zeroLngLat);
913 		return transform;
914 	},
915 	
916 	/**
917 	 * Convert the given longitude/latitude to returned [x, y] coordinates.
918 	 * Returns null if the lng/lat is out of bounds.
919 	 * NOTE: the order of the parameters is longitude, latitude, corresponding
920 	 * with x and y.
921 	 * In order to align to the viewport, the return value should be subtracted
922 	 * from this.zeroPx (see toViewport).
923 	 * @public
924 	 * @param {Number} lng longitude
925 	 * @param {Number} lat latitude
926 	 * @return {Array[x,y]} Global projected pixels of the given lng/lat
927 	 */
928 	toPixels: function(lng, lat) {
929 		var xy=this.prj.forward(lng, lat), resolution=this.res;
930 		if (!xy) return null;
931 		xy[0]/=resolution;
932 		xy[1]/=resolution;
933 		
934 		return xy;
935 	},
936 	
937 	/**
938 	 * Return surface coordinates of the given lat/lng (offset relative to
939 	 * the zero point)
940 	 * @public
941 	 * @param {Number} lng longitude
942 	 * @param {Number} lat latitude
943 	 * @return {Array[x,y]} Surface pixels of the given lng/lat
944 	 */
945 	toSurface: function(lng, lat) {
946 		var xy=this.toPixels(lng, lat);
947 		if (!xy) return null;
948 		xy[0]-=this.zpx[0];
949 		xy[1]=this.zpx[1] - xy[1]; // Note Y axis inversion
950 		return xy;
951 	},
952 	
953 	/**
954 	 * Convert from global pixel coordinates to lng/lat.
955 	 * @public
956 	 * @param {Number} x
957 	 * @param {Number} y
958 	 * @return {Array[lng,lat]} Lng/lat corresponding to the global pixels
959 	 
960 	 */
961 	fromPixels: function(x, y) {
962 		var resolution=this.res;
963 		return this.prj.inverse(x * resolution, y * resolution);
964 	},
965 	
966 	/**
967 	 * Convert from surface coordinates to [lng, lat] offset from the zero point.
968 	 * @public
969 	 * @param {Number} x
970 	 * @param {Number} y
971 	 * @return {Array[lng,lat]} Lng/lat corresponding to the suface pixels
972 	 */
973 	fromSurface: function(x, y) {
974 		return this.fromPixels(x+this.zpx[0], this.zpx[1] - y);	// Note Y axis inversion
975 	}
976 };

980  * @name nanomaps.Projections

982 var Projections={
983 	/**
984 	 * Standard "Web Mercator" projection as used by Google, Microsoft, OSM, et al.
985 	 * Instantiate as new WebMercator(options). 
986 	 * @class
987 	 * @name nanomaps.Projections.WebMercator
988 	 * @param {Number} [options.MIN_LEVEL=1]
989 	 * @param {Number} [options.MAX_LEVEL=18]
990 	 */
991 	WebMercator: function(options) {
992 		if (!options) options={};
993 		/**
994 		 * Default center for this projection
995 		 * @public
996 		 * @name DEFAULT_CENTER
997 		 * @memberOf nanomaps.Projections.WebMercator#
998 		 */
999 		this.DEFAULT_CENTER={lat:39.7406, lng:-104.985441};
1000 		/**
1001 		 * Default resolution for this projection
1002 		 * @public
1003 		 * @name DEFAULT_RESOLUTION
1004 		 * @memberOf nanomaps.Projections.WebMercator#
1005 		 */
1006 		this.DEFAULT_RESOLUTION=611.4962;
1007 	
1008 		var EARTH_RADIUS=6378137.0,
1009 			DEG_TO_RAD=.0174532925199432958,
1010 			RAD_TO_DEG=57.29577951308232,
1011 			FOURTHPI=0.78539816339744833,
1012 			HALFPI=1.5707963267948966,
1013 			HIGHEST_RES=78271.5170;
1014 		
1015 		/**
1016 		 * Minimum valid level
1017 		 * @public
1018 		 * @name MIN_LEVEL
1019 		 * @memberOf nanomaps.Projections.WebMercator#
1020 		 */
1021 		this.MIN_LEVEL=options.MIN_LEVEL||1;
1022 		/**
1023 		 * Maximum valid level
1024 		 * @public
1025 		 * @name MAX_LEVEL
1026 		 * @memberOf nanomaps.Projections.WebMercator#
1027 		 */
1028 		this.MAX_LEVEL=options.MAX_LEVEL||18;
1029 		
1030 		/**
1031 		 * Convert from global to projected units 
1032 		 * @public
1033 		 * @name forward
1034 		 * @methodOf nanomaps.Projections.WebMercator#
1035 		 * @param {Number} x
1036 		 * @param {Number} y
1037 		 * @return {Array[x,y]}
1038 		 */
1039 		this.forward=function(x, y) {
1040 			return [
1041 				x*DEG_TO_RAD * EARTH_RADIUS, 
1042 				Math.log(Math.tan(FOURTHPI + 0.5 * DEG_TO_RAD * y)) * EARTH_RADIUS 
1043 			];
1044 		};
1045 		
1046 		/**
1047 		 * Convert from projected to global units 
1048 		 * @public
1049 		 * @name inverse
1050 		 * @methodOf nanomaps.Projections.WebMercator#
1051 		 * @param {Number} x
1052 		 * @param {Number} y
1053 		 * @return {Array[x,y]}
1054 		 */
1055 		this.inverse=function(x, y) {
1056 			return [
1057 				RAD_TO_DEG * x / EARTH_RADIUS,
1058 				RAD_TO_DEG * (HALFPI - 2. * Math.atan(Math.exp(-y/EARTH_RADIUS)))
1059 			];
1060 		};
1061 		
1062 		/**
1063 		 * Return the resolution (m/px) for the given zoom index given a standard
1064 		 * power of two zoom breakdown.
1065 		 * @public
1066 		 * @name fromLevel
1067 		 * @methodOf nanomaps.Projections.WebMercator#
1068 		 * @param {Number} level
1069 		 * @return {Number} resolution
1070 		 */
1071 		this.fromLevel=function(level) {
1072 			return HIGHEST_RES/Math.pow(2, level-1);
1073 		};
1074 		
1075 		/**
1076 		 * Return the level for the given resolution on the standard power of
1077 		 * two scale.
1078 		 * @public
1079 		 * @name toLevel
1080 		 * @methodOf nanomaps.Projections.WebMercator#
1081 		 * @param {Number} resolution
1082 		 * @return {Number} level
1083 		 */
1084 		this.toLevel=function(resolution) {
1085 			return Math.log(HIGHEST_RES/resolution) / Math.log(2) + 1;
1086 		};
1087 		
1088 		// release memory from closure
1089 		options=null;
1090 	}
1091 };
