1 /**
  2  * Extend MapSurface with standard support for desktop event handling:
  3  *  - mouse events
  4  *  - mouse cursors
  5  *  - translation of desktop events to gestures
  6  */
  7 (function(nanomaps) {
  8 	var DOUBLE_CLICK_TIME=100;
  9 	
 10 	var MapSurfaceMethods=nanomaps.MapSurface.prototype;
 11 	
 12 	// State management
 13 	function updateMouseState(map, action, event) {
 14 		var mouseState=map._mouseState, coords=map.eventToContainer(event),
 15 			deltaX, deltaY;
 16 			
 17 		switch (action) {
 18 		case 'down':
 19 			// A down click resets everything
 20 			clearMouseState(map);
 21 			mouseState.s='down';
 22 			mouseState.start=coords;
 23 			if (!mouseState.ccount) mouseState.ccount=0;
 24 
 25 			if (mouseState.clickTimer) {
 26 				clearTimeout(mouseState.clickTimer);
 27 				mouseState.clickTimer=null;
 28 			}
 29 
 30 			break;
 31 			
 32 		case 'up':
 33 			if (mouseState.s=='down') {
 34 				// Simulate a click
 35 				mouseState.ccount+=1;
 36 				if (mouseState.clickTimer) {
 37 					clearTimeout(mouseState.clickTimer);
 38 					mouseState.clickTimer=null;
 39 				}
 40 				
 41 				if (mouseState.ccount>1) {
 42 					// trigger double click
 43 					//console.log('double click');
 44 					mouseState.ccount=0;
 45 					
 46 					deltaY=Math.ceil(map.getLevel()+0.1);
 47 					map.setLevel(deltaY, coords);
 48 				} else {
 49 					mouseState.clickTimer=setTimeout(function() {
 50 						mouseState.clickTimer=null;
 51 						
 52 						// single click
 53 						//console.log('single click: lat=' + map.toLatLng(coords.x, coords.y).lat + ', lng=' + map.toLatLng(coords.x, coords.y).lng);
 54 						mouseState.ccount=0;
 55 					}, DOUBLE_CLICK_TIME);
 56 				}
 57 			} 
 58 			// Just clear the state
 59 			clearMouseState(map);
 60 			break;
 61 			
 62 		case 'move':
 63 			if (mouseState.s==='down'||mouseState.s==='drag') {
 64 				mouseState.s='drag';
 65 				if (!mouseState.last) mouseState.last=mouseState.start;
 66 				deltaX=mouseState.last.x-coords.x;
 67 				deltaY=mouseState.last.y-coords.y;
 68 				mouseState.last=coords;
 69 				
 70 				map.moveBy(deltaX, -deltaY);
 71 			}
 72 			break;
 73 		}
 74 		
 75 	}
 76 	function clearMouseState(map) {
 77 		var mouseState=map._mouseState;
 78 		mouseState.s=null;
 79 		mouseState.start=null;
 80 		mouseState.last=null;
 81 	}
 82 	
 83 	
 84 	// Event handlers
 85 	MapSurfaceMethods.ondom_mousedown=function(event) {
 86 		updateMouseState(this, 'down', event);
 87 		event.preventDefault();
 88 	};
 89 	MapSurfaceMethods.ondom_mouseup=function(event) {
 90 		if (!this._mouseState.s) return;	// Only process a move if in a state
 91 
 92 		updateMouseState(this, 'up', event);
 93 		event.preventDefault();
 94 	};
 95 	MapSurfaceMethods.ondom_mousemove=function(event) {
 96 		if (!this._mouseState.s) return;	// Only process a move if in a state
 97 		
 98 		updateMouseState(this, 'move', event);
 99 		event.preventDefault();
100 	};
101 	MapSurfaceMethods.ondom_mousewheel=function(event) {
102 		clearMouseState(this);
103 		event.preventDefault();
104 		
105 		// Handle scroll (TODO - this needs to zoom around the mouse point)
106 		var delta, factor=10, level=Math.round(this.getLevel()*factor)/factor,
107 			coords=map.eventToContainer(event);
108 		if (event.wheelDelta) delta=event.wheelDelta/120;	// IE
109 		else if (event.delta) delta=-event.delta/3;
110 		else delta=-event.detail/3;
111 		
112 		this.setLevel(level+delta/10.0, coords);
113 	};
114 	
115 	// Attach to initialization
116 	MapSurfaceMethods.advise('initialize', 'after', function(options) {
117 		this._mouseState={};
118 		this.routeDomEvent('mousedown', null, 'glass');
119 		this.routeDomEvent('mouseup', null, 'document');
120 		this.routeDomEvent('mousemove', null, 'document');
121 		if (!options.disableMouseWheel) {
122 			this.routeDomEvent('DOMMouseScroll', 'dom_mousewheel', 'glass');
123 			this.routeDomEvent('mousewheel', null, 'glass');
124 		}
125 		this.elements.viewport.style.cursor='move';
126 	});
127 })(nanomaps);
